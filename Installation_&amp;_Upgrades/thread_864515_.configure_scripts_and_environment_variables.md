export SPEEX_CFLAGS=" -I$PWD/../speex/include -I../speex/include/speex"
export SPEEX_LIBS="$PWD/../speex/libspeex/libspeex.la"
./configure

This method works correctly on e.g. Zenwalk, but not on Kubuntu.
