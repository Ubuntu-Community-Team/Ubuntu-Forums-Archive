### Post by Aetherius on 2007-09-05
Don't know if this is any help but I just thought I post my experiences. 

My intel 945 was quite easy to configure for dual screening, but this wasn't enough for me as I wanted composite on both screens.

So with a bit of research and some luck I found out about MergedFB, which allowed me to *technically* output a resolution of 2048*768 and spread it across boths screens! This works brilliantly, and metacity/kwin still treats the monitors as seperate (as far as maximizing etc goes). Now my Xorg.conf looks like this. (without all the irrelevant stuff)

```
Section "Device"
        Identifier      "Intel Corporation Mobile Integrated Graphics Controller"
        Driver          "i810"
        BusID           "PCI:0:2:0"

        Option  "DRI"                           "true"
        Option  "MergedFB"                      "true"
        Option  "MonitorLayout"                 "CRT,LFP"

        Option  "SecondPosition"                "RightOf"
        Option  "MetaModes"                     "1024x768-1024x768"
        Option  "SecondMonitorHorizSync"        "28-49"
        Option  "SecondMonitorVertRefresh"      "60-72"
        Option  "MergedNonRectangular"          "true"
EndSection

Section "Monitor"
        Identifier      "Generic Monitor"
        Option          "DPMS"
EndSection

Section "Screen"
        Identifier      "Default Screen"
        Device          "Intel Corporation Mobile Integrated Graphics Controller"
        Monitor         "Generic Monitor"
        DefaultDepth    24
        SubSection "Display"
                Depth           24
                Modes           "1600x1200" "1280x1024" "1280x800" "1024x768" "800x600" "640x480" "320x240"
        EndSubSection
EndSection

Section "ServerLayout"
        #Option "AIGLX" "true"
        Identifier      "Default Layout"
        Screen          "Default Screen"
        InputDevice     "Generic Keyboard"
        InputDevice     "Configured Mouse"
        InputDevice     "Synaptics Touchpad"
EndSection

Section "DRI"
        Mode    0666
EndSection

Section "Extensions"
        Option "Composite" "Enable"
EndSection
```
