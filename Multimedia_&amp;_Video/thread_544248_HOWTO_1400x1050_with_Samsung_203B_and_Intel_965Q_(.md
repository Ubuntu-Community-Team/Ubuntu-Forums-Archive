### Post by oh6eh on 2007-09-06
[**This**]("http://ubuntuforums.org/showthread.php?t=451719") exellent HOWTO helped me to get native resolution with this monitor. Refer it for more detailed instructions.

Key ingredients:[LIST]
[*]i810 driver (instead if intel)
[*]915resolution patching for ALL depths (8, 16, 32)
[*]VertRefresh	60.0 (exact figure)
[*]Modeline	"1400x1050" 121.8 1400 1488 1632 1864 1050 1053 1057 1089

[/LIST]

Create a new file:
/etc/init.d/samsung:
```

#!/bin/sh
/usr/sbin/915resolution 3a 1400 1050 8 1864 1089
/usr/sbin/915resolution 4b 1400 1050 16 1864 1089
/usr/sbin/915resolution 5a 1400 1050 32 1864 1089

```
(This is a kind of quick hack but it works)

Run
```

# cd /etc/init.d
# update-rc.d samsung start 99 S .

```

Finally edit your /etc/X11/xorg.conf to resemble this:

```

	Driver		"wacom"
	Identifier	"cursor"
	Option		"Device"	"/dev/input/wacom"
	Option		"Type"		"cursor"

Section "Device"
	Identifier	"Intel Corporation 82945G/GZ Integrated Graphics Controller"
	Driver		"i810"
	BusID		"PCI:0:2:0"
EndSection

Section "Monitor"
	Identifier	"SyncMaster"
	DisplaySize	408 300
	HorizSync	30.0 - 81.0
	VertRefresh	60.0
	Option		"DDC" "false"
	Modeline	"1400x1050" 121.8 1400 1488 1632 1864 1050 1053 1057 1089
	Option		"DPMS"
#	HorizSync	30-81
#	VertRefresh	56-75
EndSection

Section "Screen"
	Identifier	"Default Screen"
	Device		"Intel Corporation 82945G/GZ Integrated Graphics Controller"
	Monitor		"SyncMaster"
	DefaultDepth	24
	SubSection "Display"
		Depth		8
		Modes		"1400x1050"
	EndSubSection
	SubSection "Display"
		Depth		16
		Modes		"1400x1050"
	EndSubSection
	SubSection "Display"
		Depth		24
		Modes		"1400x1050"
	EndSubSection
EndSection

Section "ServerLayout"
	Identifier	"Default Layout"
	Screen		"Default Screen"
	InputDevice	"Generic Keyboard"
	InputDevice	"Configured Mouse"
	InputDevice     "stylus"	"SendCoreEvents"
	InputDevice     "cursor"	"SendCoreEvents"
	InputDevice     "eraser"	"SendCoreEvents"
```
