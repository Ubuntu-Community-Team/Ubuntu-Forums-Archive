### Post by Ryba on 2007-04-02
I don't know if this is your problem or not but not all ati video cards work with ati drivers (fglrx). Sounds stupid but its true.

In my case, I have a thinkpad T42 and the ati video card on it (if you look at ati's website) is specifically listed as a video card not supported by ati's fglrx drivers. They highly recommend you not use their drivers for their video cards if their drivers are not supported.

Because of that, I have to use something similiar to this (radeon driver but still very good 3d acceleration)

```
Section "Files"
        FontPath        "/usr/share/X11/fonts/misc"
        FontPath        "/usr/share/X11/fonts/cyrillic"
        FontPath        "/usr/share/X11/fonts/100dpi/:unscaled"
        FontPath        "/usr/share/X11/fonts/75dpi/:unscaled"
        FontPath        "/usr/share/X11/fonts/Type1"
        FontPath        "/usr/share/X11/fonts/100dpi"
        FontPath        "/usr/share/X11/fonts/75dpi"
        # path to defoma fonts
        FontPath        "/var/lib/defoma/x-ttcidfont-conf.d/dirs/TrueType"
EndSection

Section "Module"
        Load    "i2c"
        Load    "bitmap"
        Load    "ddc"
        Load    "dri"
        Load    "dbe" # Was not here before
        Load    "extmod"
        Load    "freetype"
        Load    "glx"
        Load    "int10"
        Load    "type1"
        Load    "vbe"
EndSection

Section "InputDevice"
        Identifier      "Generic Keyboard"
        Driver          "kbd"
        Option          "CoreKeyboard"
        Option          "XkbRules"      "xorg"
        Option          "XkbModel"      "pc104"
        Option          "XkbLayout"     "us"
EndSection

Section "InputDevice"
        Identifier      "Configured Mouse"
        Driver          "mouse"
        Option          "CorePointer"
        Option          "Device"                "/dev/input/mice"
        Option          "Protocol"              "ExplorerPS/2"
        Option          "ZAxisMapping"          "4 5"
        Option          "Emulate3Buttons"       "true"
EndSection

        Identifier      "Synaptics Touchpad"
        Driver          "synaptics"
        Option          "SendCoreEvents"        "true"
        Option          "Device"                "/dev/psaux"
        Option          "Protocol"              "auto-dev"
        Option          "HorizScrollDelta"      "0"
EndSection

Section "Device"
        Identifier      "Radeon Mobility M7"
        Driver          "radeon"
        BusID           "PCI:1:0:0"
        Option          "DynamicClocks" "on"
        Option          "MergedFB"      "true"

        # The next line lets me switch between dual-head and several clone modes of varying resolutions with xrandr.

        #Option         "CRT2HSync"     "30-81"

        #Screen         0
        Option          "XAANoOffscreenPixmaps" "true"

EndSection
```
