### Post by dchey on 2007-12-12
Hi barney_1

This is the entry under modules form my xorg.conf that is working the best for me !


Section "Device"
	Identifier	"ATI Technologies Inc Radeon R100 QD [Radeon 7200]"
	Driver		"ati"
	BusID		"PCI:1:0:0"
	Option		"DRI"			"true"
	Option		"ColorTiling"		"on"
	Option		"EnablePageFlip"	"true"
	Option		"AccelMethod" 		"XAA"
	Option 		"AccelDFS" 		"true"
	Option		"XAANoOffscreenPixmaps"
	Option		"RenderAccel"		"true"
	Option 		"AGPMode" 		"4"
	Option		"AGPFastWrite"		"on"
	Option		"sync-video"		"off"
	Option		"AGPSize" 		"64" 
	Option		"EnableDepthMoves" 	"true"
        Option          "TripleBuffer"   	"true"
        Option          "GARTSize"       	"64"
EndSection
"
The Driver needs to be set to "ati" to work.
