### Post by myk.dinis on 2010-05-07
Here they are... :)

# Firewall Config File (From Scratch)
# delete all existing rules.
iptables -F
iptables -t nat -F
iptables -t mangle -F
iptables -X

#Default Policy Settings
iptables -P INPUT DROP
iptables -P FORWARD DROP

# Prerouting VPN and Web Exceptions
iptables -t nat -A PREROUTING -i eth2 -p udp -d 192.168.10.254 --dport 4500 -j DNAT --to 10.1.1.4:4500
iptables -t nat -A PREROUTING -i eth2 -p udp -d 192.168.10.254 --dport 500 -j DNAT --to 10.1.1.4:500
iptables -t nat -A PREROUTING -i eth2 -p 50 -d 192.168.10.254 -j DNAT --to 10.1.1.4
iptables -t nat -A PREROUTING -i eth0 -p tcp -d 192.168.10.253 --dport 80 -j DNAT --to 10.1.2.2:80

# Always accept loopback traffic
iptables -A INPUT -i lo -j ACCEPT

# Allow established connections, and those not coming from the outside
iptables -A INPUT -m conntrack --ctstate ESTABLISHED,RELATED -j ACCEPT
iptables -A FORWARD -i eth0 -o eth1 -m conntrack --ctstate ESTABLISHED,RELATED -j ACCEPT
iptables -A FORWARD -i eth2 -o eth3 -m conntrack --ctstate ESTABLISHED,RELATED -j ACCEPT
iptables -A FORWARD -i eth1 -o eth3 -m conntrack --ctstate ESTABLISHED,RELATED -j ACCEPT

# Input VPN and Web Exceptions (Ext to Int)
iptables -A INPUT -p udp -i eth3 --dport 4500 -j ACCEPT
iptables -A INPUT -p udp -i eth3 --dport 500 -j ACCEPT
iptables -A INPUT -p 50 -i eth3 -j ACCEPT
iptables -A INPUT -p tcp -i eth1 --dport 80 -j ACCEPT

# Allow outgoing connections from the LAN side.
iptables -A FORWARD -i eth1 -o eth0 -j ACCEPT
iptables -A FORWARD -i eth3 -o eth2 -j ACCEPT

# Masquerade.
iptables -t nat -A POSTROUTING -o eth0 -j MASQUERADE
iptables -t nat -A POSTROUTING -o eth2 -j MASQUERADE

# Don't forward from the outside to the inside.
#iptables -A FORWARD -i eth0 -o eth0 -j DROP
#iptables -A FORWARD -i eth2 -o eth2 -j DROP

# Enable routing.
echo 1 > /proc/sys/net/ipv4/ip_forward
