Script for the traffic shaping.
```
#!/bin/bash

#Traffic Shapping (QoS) script by Credomane
#source for robbed parts: http://lartc.org/howto/index.html

#In kilobits
DEV=eth0
DOWNLINK=2048
UPLINK=256

function stop {
	# clean existing down- and uplink qdiscs, hide errors
	tc qdisc del dev $DEV root    2> /dev/null > /dev/null
	tc qdisc del dev $DEV ingress 2> /dev/null > /dev/null
}


function start {
	########## uplink #############
	# slow uploads down to somewhat less than the real speed to prevent 
	# queuing at our modem/ISP. Tune to see how high you can set it.
	#
	tc qdisc add dev $DEV root handle 1: htb default 40
	tc class add dev $DEV parent 1: classid 1:1 htb rate ${UPLINK}kbit ceil ${UPLINK}kbit burst 6k

	#ULTRA	= dns, ack packets.
	#HIGH	= games.
	#MEDIUM	= ssh, http, https, webmin.
	#LOW		= default class
	#BULK	= bittorrent and crap like that.
	tc class add dev $DEV parent 1:1 classid 1:10 htb rate $[6*$UPLINK/10]kbit ceil $[7*$UPLINK/10]kbit burst 6k prio 1
	tc class add dev $DEV parent 1:1 classid 1:20 htb rate $[8*$UPLINK/10]kbit ceil $[9*$UPLINK/10]kbit burst 6k prio 2
	tc class add dev $DEV parent 1:1 classid 1:30 htb rate $[6*$UPLINK/10]kbit ceil $[7*$UPLINK/10]kbit burst 6k prio 3
	tc class add dev $DEV parent 1:1 classid 1:40 htb rate $[6*$UPLINK/10]kbit ceil $[7*$UPLINK/10]kbit burst 6k prio 4
	tc class add dev $DEV parent 1:1 classid 1:50 htb rate $[6*$UPLINK/10]kbit ceil $[7*$UPLINK/10]kbit burst 6k prio 5

	tc qdisc add dev $DEV parent 1:10 handle 110: sfq perturb 10
	tc qdisc add dev $DEV parent 1:20 handle 120: sfq perturb 10
	tc qdisc add dev $DEV parent 1:30 handle 130: sfq perturb 10
	tc qdisc add dev $DEV parent 1:40 handle 140: sfq perturb 10
	tc qdisc add dev $DEV parent 1:50 handle 150: sfq perturb 10

	#Create the sorting filters.
	#Use iptables mangle preroute table to "MARK" packets.
	tc filter add dev $DEV parent 1: protocol ip prio 1 handle 1 fw classid 1:10
	tc filter add dev $DEV parent 1: protocol ip prio 2 handle 2 fw classid 1:20
	tc filter add dev $DEV parent 1: protocol ip prio 3 handle 3 fw classid 1:30
	tc filter add dev $DEV parent 1: protocol ip prio 4 handle 4 fw classid 1:40
	tc filter add dev $DEV parent 1: protocol ip prio 5 handle 5 fw classid 1:50

	########## downlink #############
	# slow downloads down to somewhat less than the real speed to prevent 
	# queuing at our ISP. Tune to see how high you can set it.
	# ISPs tend to have *huge* queues to make sure big downloads are fast
	#
	# attach ingress policer:
	tc qdisc add dev $DEV handle ffff: ingress

	# filter *everything* to it (0.0.0.0/0), drop everything that's
	# coming in too fast:
	tc filter add dev $DEV parent ffff: protocol ip prio 50 u32 match ip src \
		0.0.0.0/0 police rate ${DOWNLINK}kbit burst 12k drop flowid :1
}

function status {
	tc -s class show dev $DEV
}

case $1 in
	"start")
		start;
	;;
	"stop")
		stop;
	;;
	"restart")
		stop;
		start;
	;;
	"status")
		status;
	;;
	*)
		echo "Usage: $0 start|stop|restart|status";
	;;
esac




exit 1;

```
