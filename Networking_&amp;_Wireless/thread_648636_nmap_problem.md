### Post by Elitist_Phoenix on 2007-12-25
root@scorpion:/home/scorpion# cat /etc/resolv.conf 
# generated by NetworkManager, do not edit!

nameserver 192.168.1.1


Alternatively
root@scorpion:/home/scorpion# ifconfig 
eth0      Link encap:Ethernet  HWaddr 00:... 
          inet addr:192.168.1.10  Bcast:192.168.1.255  Mask:255.255.255.0

inet addr:192.168.1.10 <== Change the last numeral after the last dot to a 1


nmap --dns_servers 192.168.1.1 [www.host-to-scan.com](www.host-to-scan.com)
