### Post by Richard_Thornton on 2015-05-14
This config is basic, your interfaces are probably different, this works for me after reboot, ExecStop doesn't work, it looks like it needs the PID as an argument.

sudo vi /etc/teamd.conf


{
 "device": "team0",
 "runner": {"name": "roundrobin"},
 "ports": {"p2p1": {}, "p2p2": {}}
}




sudo vi /lib/systemd/system/teamd.service 


[Unit]
Description=Custom Start and Shutdown


[Service]
Type=oneshot
ExecStart=/usr/bin/teamd -d -f /etc/teamd.conf
ExecStop=/usr/bin/teamd -k team0
RemainAfterExit=yes


[Install]
WantedBy=multi-user.target


sudo systemctl enable teamd.service



sudo vi /etc/network/interfaces


auto lo
iface lo inet loopback


auto team0 
iface team0 inet dhcp
