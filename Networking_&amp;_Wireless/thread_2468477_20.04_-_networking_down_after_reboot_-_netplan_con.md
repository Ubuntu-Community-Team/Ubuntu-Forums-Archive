```

  ethernets:

      dhcp4: false
      dhcp6: false

      optional: true

      dhcp4: false
      dhcp6: false

    bond0:
      interfaces: [enp1s0f0, enp1s0f1]
      optional: true

      gateway4: 10.10.10.145
```
