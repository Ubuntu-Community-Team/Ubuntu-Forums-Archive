I didn't say anything about audio problems.  There were though; all of my Eee PC users were stuck without proper mute switch functionality in Jaunty.  The mute switch wasn't properly implemented for their audio driver so when mute was toggled with the hotkey users had popping and hissing from their speakers rather than silence.  There were tons of complaints about that one.

The work around was to disable the gnome shortcut for mute and volume and to use the following as replacements (it's REALLY ugly, but it worked):

```

  $KEY_VOLU)
     if [ "$VOL_DEV" = "AUTO" ]; then
       VOL_DEVS=($(amixer -c 0 controls | awk -F "," ' $_ ~ /Playback Volume/ \
	{split($3,a,"=");print a[2]}' | sed -s -e "s#\sPlayback\sVolume'##" -e "s#^'##"))

       for dev in ${VOL_DEVS[*]}; do
         amixer -q -c 0 set $dev $VOL_ADJ+ >/dev/null 2>&1
       done
     elif [ "$VOL_DEV" = "FAKE" ]; then
       acpi_fakekey 115
     else
       amixer -q -c 0 set $VOL_DEV $VOL_ADJ+ &
     fi
  ;;
  $KEY_VOLD)
     if [ "$VOL_DEV" = "AUTO" ]; then
       VOL_DEVS=($(amixer -c 0 controls | awk -F "," ' $_ ~ /Playback Volume/ \
	{split($3,a,"=");print a[2]}' | sed -s -e "s#\sPlayback\sVolume'##" -e "s#^'##"))

       for dev in ${VOL_DEVS[*]}; do
         amixer -q -c 0 set $dev $VOL_ADJ- >/dev/null 2>&1
       done
     elif [ "$VOL_DEV" = "FAKE" ]; then
       acpi_fakekey 114
     else
       amixer -q -c 0 set $VOL_DEV $VOL_ADJ- &
     fi
  ;;
  $KEY_MUTE)
     if [ "$VOL_DEV" = "AUTO" ]; then
       MUTE_DEVS=($(amixer -c 0 controls | grep "Playback Switch" | awk -F "," '{printf $1" "}'))
       MUTE_STATE=$(amixer -c 0 cget ${MUTE_DEVS[0]},iface=MIXER | awk '$_ ~ /:/ {split($2,a,"="); print a[2]}')
       if [[ "$MUTE_STATE" =~ "on,on" ]]; then
         for dev in ${MUTE_DEVS[*]}; do
           amixer -q -c 0 cset $dev,iface=MIXER 0 >/dev/null 2>&1
         done
       else
         for dev in ${MUTE_DEVS[*]}; do
           amixer -q -c 0 cset $dev,iface=MIXER 1 >/dev/null 2>&1
         done
       fi
     fi
  ;;

```
