if yuo liked the old nautilus in gnome 2 you may like [caja]("http://mate-desktop.org/install/#ubuntu"), just add the repo and install the app you want, it has a few dependencies, but it works, i am using xfce 4.10 and had a issue with a applet opening thunar so i moved thunar to thunar.bak and made a script called thunar
```
#!/bin/bash
if [ -n "$2" ]; then
    if [ -n "$3" ]; then
        thunar.bak "$1" "$2" "$3"
    else
        thunar.bak "$1" "$2"
    fi
elif [ -z "$1" ]; then
    thunar.bak
elif [ -d "$1" ]; then
    caja "$1" &
else
    thunar.bak "$1"
fi
exit
