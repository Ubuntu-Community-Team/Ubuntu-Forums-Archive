### Post by iaculallad on 2008-06-06
The lines of text below is an edited version of your xorg.conf file.

First, backup your current /etc/X11/xorg.conf file

> sudo cp /etc/X11/xorg.conf /etc/X11/xorg.conf.original

Second, open your /etc/X11/xorg.conf file for editing

> sudo gedit /etc/X11/xorg.conf

Third, select all text inside the xorg.conf and delete it. Then copy and paste the lines below to your blank xorg.conf file and save it. Restart.

```
# xorg.conf (X.Org X Window System server configuration file)
#
# This file was generated by failsafeDexconf, using
# values from the debconf database and some overrides to use vesa mode.
#
# You should use dexconf or another such tool for creating a "real" xorg.conf
# For example:
#   sudo dpkg-reconfigure -phigh xserver-xorg
Section "InputDevice"
    Identifier    "Generic Keyboard"
    Driver        "kbd"
    Option        "XkbRules"    "xorg"
    Option        "XkbModel"    "pc105"
    Option        "XkbLayout"    "us"
EndSection

Section "InputDevice"
    Identifier    "Configured Mouse"
    Driver        "mouse"
    Option        "Emulate3Buttons"    "true"
EndSection

Section "Device"
    Identifier    "Configured Video Device"
    Boardname    "vesa"
    Busid        "PCI:1:0:0"
    Driver        "nvidia"
    Screen    0
    Option        "NoLogo"    "True"
       EndSection

Section "Monitor"
    Identifier    "Configured Monitor"
    Vendorname    "Plug 'n' Play"
    Modelname    "Plug 'n' Play"
    HorizSync 30-110
    VertRefresh 50-160
EndSection

Section "Screen"
    Identifier    "Default Screen"
    Device        "Configured Video Device"
    Monitor        "Configured Monitor"
    DefaultDepth 24
                SubSection "Display"
                Depth 24
                Modes "1280x1024"
    EndSubSection
EndSection

Section "ServerLayout"
    Identifier    "Default Layout"
  screen 0 "Default Screen" 0 0
EndSection
Section "Module"
    Load        "glx"
    Load        "v4l"
EndSection
Section "ServerFlags"
EndSection  
```
