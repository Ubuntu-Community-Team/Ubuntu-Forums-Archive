### Post by Claus7 on 2019-07-18
Hello,

copying the entire folder script to script 2 and changing the file script.script to:
# This is an example plymouth plugin script

Window.SetBackgroundTopColor(0.234, 0.93, 0.705);
Window.SetBackgroundBottomColor(0.16, 0.85, 0.44);

logo.image = Image("flower1.png");
logo.sprite = Sprite(logo.image);
logo.opacity_angle = 0;

flower2.image = Image("flower2.png");
flower2.sprite = Sprite(flower2.image);

flower2.sprite.SetX(-10);
flower2.sprite.SetY(20);
flower2.sprite.SetZ(10);

fun refresh_callback ()
  {
    if (status == "normal")
      {
        logo.opacity_angle += ((2 * 3.14) / 50) * 0.5;  # 0.5 HZ
        min_opacity = 0.3;
        opacity = (Math.Cos(logo.opacity_angle) + 1) / 2;
        opacity *= 1 - min_opacity;
        opacity += min_opacity;
        logo.sprite.SetX (Window.GetX() + Window.GetWidth()  / 2 - logo.image.GetWidth()  / 2);
        logo.sprite.SetY (Window.GetY() + Window.GetHeight() / 2 - logo.image.GetHeight() / 2);
        logo.sprite.SetOpacity (opacity);
      }
    else
      {
        logo.sprite.SetX (0);
        logo.sprite.SetY (0);
        logo.sprite.SetOpacity (1);
      }
  }
  
Plymouth.SetRefreshFunction (refresh_callback);

....

the rest remains as the original, I was able to see two flower pictures, yet trying to add animation did not work out.
edit: I think that playing with the opacity of the images might be what you are looking for. The other alternative would be the animation of images coming back and forth at a specific time interval.
