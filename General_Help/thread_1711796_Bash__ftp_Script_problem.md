### Post by Don Barnett on 2011-03-21
The Cron is:
15,45 5-23 * * * /var/www/home/GetOrdersFtp >> /var/www/home/logs/GetOrders_log.dat

GetOrderFtp:
#/bin/bash
# This script executs an ftp script for transferring files from a remote host
# It then checks the results of the transfer against a list of files from the remote.
# If all  of the files size are the same then the transfer is considered a success.
# A second ftp script then executes which removes the files from the remote host
# that where just transfered.

date
if [ -e /var/www/home/htsdata/FTPcompareList.txt ]; then
	rm /var/www/home/htsdata/FTPcompareList.txt
fi
if [ -e /var/www/home/htsdata/Archive.ftp ]; then
	rm /var/www/home/htsdata/Archive.ftp
fi
if [ -e /var/www/home/htsdata/delete.ftp ]; then
	rm /var/www/home/htsdata/delete.ftp
fi
while true; do
	# change this next line to match the proper ftp script to execute
	ftp -v < /var/www/home/ftpScripts/GetOrdersFtpScript
	filename=/var/www/home/htsdata/FTPcompareList.txt
	filesize=$(stat -c%s "$filename")
	echo "$filesize"
# this loop compare each file recieved with the files in the FTPcompareList.txt generated by the remote ftp
	if [ $filesize -gt 0 ]; then
		cd /var/www/home/htsdata/Orders
		TestResult=1
		while read tp tx us gp sz dt tm fn; do
			lfn=`expr length "$fn"`
			if [ $lfn -gt 0 ]; then
				echo "$fn"
				xline=`grep "$fn"  /var/www/home/htsdata/FTPcompareList.txt`
				echo "$xline"
				size=`expr substr "$xline" 35 7`
				echo "$sz"
				echo ${size# *}
				if [ $sz -eq ${size# *} ]; then
					echo "equal"
				else
					echo "Not equal"
					TestResult=0
				fi
			fi
		echo $TestResult
		done < <(ls -l)
		if [ $TestResult -eq 1 ]; then
			>> /var/www/home/htsdata/Archive.ftp echo "remote.com"
			>> /var/www/home/htsdata/Archive.ftp echo "prompt off"
			>> /var/www/home/htsdata/Archive.ftp echo "cd Orders/Archive"
			>> /var/www/home/htsdata/s.ftp echo "binary"
			while read filen; do
				>> /var/www/home/htsdata/Archive.ftp echo put "$filen"
			done < <(ls -1 *.txt)
			>> /var/www/home/htsdata/Archive.ftp echo "bye"
			ftp -v < /var/www/home/htsdata/Archive.ftp
			echo "Archive Transfer - Finished"

			>> /var/www/home/htsdata/delete.ftp echo "remote.com"
			>> /var/www/home/htsdata/delete.ftp echo "prompt off"
			>> /var/www/home/htsdata/delete.ftp echo "cd Orders"
			>> /var/www/home/htsdata/delete.ftp echo "binary"
			while read filen; do
				>> /var/www/home/htsdata/delete.ftp echo delete "$filen"
			done < <(ls -1 *.txt)
			>> /var/www/home/htsdata/delete.ftp echo "bye"
			ftp -v < /var/www/home/htsdata/delete.ftp
			echo "Delete Processed Completed - Finished"; break
		else
			echo "Transfer failed - Starting Over"	
			break
		fi
	else
		echo "No Files to transfer - Finished";break
	fi
done
exit


GetOrderFtpScript:

open remote.com
prompt off 
cd Orders
lcd /var/www/home/htsdata/Orders
binary
mget *.txt
ls *.txt /var/www/home/htsdata/FTPcompareList.txt
bye
