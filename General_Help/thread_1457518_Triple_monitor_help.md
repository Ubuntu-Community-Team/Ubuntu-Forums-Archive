### Post by KrisWillis on 2010-04-28
I have successful triple monitor set-ups in both my office and at home, here is my xorg.conf for my office machine - It might give you some clues...

```
Section "ServerLayout"
    Identifier     "Layout0"
    Screen      0  "Screen0" RightOf "Screen1"
    Screen      1  "Screen1" 1280 0
    Screen      2  "Screen2" LeftOf "Screen1"
    InputDevice    "Keyboard0" "CoreKeyboard"
    InputDevice    "Mouse0" "CorePointer"
EndSection

Section "Module"
    Load           "dbe"
    Load           "extmod"
    Load           "type1"
    Load           "freetype"
    Disable	   "glx"
EndSection

Section "ServerFlags"
    Option         "Xinerama" "on"
EndSection

Section "InputDevice"
    Identifier     "Keyboard0"
    Driver         "keyboard"
EndSection

Section "InputDevice"
    Identifier     "Mouse0"
    Driver         "mouse"
    Option         "Protocol" "auto"
    Option         "Device" "/dev/psaux"

    VertRefresh     43.0 - 60.0
```
