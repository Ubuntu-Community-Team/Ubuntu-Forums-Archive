### Post by KrisCarr on 2009-07-10
Try adding a prefered mode in your xorg.conf.
 
 
here is mine
 
note the Option "PreferredMode" "1920x1200_60.00" line in the monitor section.
 
this means my default is 1920x1200 resolution @ 60hz
 
 ```

Section "ServerLayout"
Identifier "aticonfig Layout"
Screen 0 "aticonfig-Screen[0]-0" 0 0
EndSection
 
Section "Files"
EndSection
 
Section "Module"
EndSection
 
Section "ServerFlags"
Option "AIGLX" "Off"
EndSection
 
Section "Monitor"
Identifier "aticonfig-Monitor[0]-0"
Option "VendorName" "ATI Proprietary Driver"
Option "ModelName" "Generic Autodetecting Monitor"
Option "DPMS" "true"
Modeline "1920x1200_60.00" 193.16 1920 2048 2256 2592 1200 1201 1204 1242 -HSync +Vsync
Option "PreferredMode" "1920x1200_60.00"
EndSection
 
Section "Device"
Identifier "aticonfig-Device[0]-0"
Driver "fglrx"
BusID "PCI:2:0:0"
Option "monitor-DVI-I_1/digital" "aticonfig-Monitor[0]-0"
 
EndSection
 
Section "Screen"
Identifier "aticonfig-Screen[0]-0"
Device "aticonfig-Device[0]-0"
Monitor "aticonfig-Monitor[0]-0"
DefaultDepth 24
SubSection "Display"
Viewport 0 0
Depth 24
EndSubSection
EndSection
 
Section "Extensions"
Option "Composite" "Off"
EndSection
 
```
I hope this helps
