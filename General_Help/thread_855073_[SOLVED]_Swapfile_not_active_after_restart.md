#
#  -- This file has been automaticly generated by ntfs-config -- 
#
# <file system> <mount point>   <type>  <options>       <dump>  <pass>

proc /proc proc defaults 0 0
# Entry for /dev/sda2 :
UUID=b15dff4f-ce6c-4d4d-8067-7286a891c8d0 / ext3 relatime,errors=remount-ro 0 1
# Entry for /dev/sda3 :
UUID=[color=red]c476259c-aa17-48e8-a009-8347decf6991[/color] none swap sw 0 0
/dev/scd0 /media/cdrom0 udf,iso9660 user,noauto,exec,utf8 0 0
/dev/sdc1 /media/GAMES\040&\040PORN ntfs-3g defaults,locale=en_GB.UTF-8 0 0
/dev/sda4 /media/PROGS\040&\040PICS ntfs-3g defaults,locale=en_GB.UTF-8 0 0

/dev/sdb1 /media/DOWNLOADS ext3 defaults 0 0
olly@Olleh:~$
```


```
olly@Olleh:~$ sudo blkid
/dev/sda1: UUID="BE98536C98532267" LABEL="WINDOWS" TYPE="ntfs" 
/dev/sda2: UUID="b15dff4f-ce6c-4d4d-8067-7286a891c8d0" TYPE="ext3" 
/dev/sda3: TYPE="swap" UUID="[color=blue]643eac00-4f10-4158-9be1-fdc8fd234254[/color]" 
/dev/sda4: UUID="00EDF97F283CD1D8" LABEL="PROGS & PICS" TYPE="ntfs" 
/dev/sdb1: LABEL="DOWNLOADS" UUID="a2bfe601-1b59-4716-bf79-9cb895fb7b3a" SEC_TYPE="ext2" TYPE="ext3" 
/dev/sdc1: UUID="36F677635F53C9C1" LABEL="GAMES & PORN" TYPE="ntfs" 
olly@Olleh:~$
```


See the difference ```
sudo cp /etc/fstab /etc/fstab.old
gksudo gedit /etc/fstab
``` and change the UUID for the swap partition to match the one given by the blkid command.Then reboot.

Good luck
