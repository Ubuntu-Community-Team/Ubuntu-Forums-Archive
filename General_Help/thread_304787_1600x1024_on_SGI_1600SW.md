### Post by scooper on 2006-11-22
Here's the relevant sections from my /etc/X11/xorg.conf.

```

Section "Screen"
    Identifier     "Default Screen"
    Device         "Generic Video Card"
    Monitor        "SGI MultiLink"
    DefaultDepth   24
    SubSection "Display"
        Depth      1
        Modes      "1600x1024" "1280x1024" "1280x960" "1024x768" "800x600" "720x400" "640x480"
    EndSubSection
    SubSection "Display"
        Depth      4
        Modes      "1600x1024" "1280x1024" "1280x960" "1024x768" "800x600" "720x400" "640x480"
    EndSubSection
    SubSection "Display"
        Depth      8
        Modes      "1600x1024" "1280x1024" "1280x960" "1024x768" "800x600" "720x400" "640x480"
    EndSubSection
    SubSection "Display"
        Depth      15
        Modes      "1600x1024" "1280x1024" "1280x960" "1024x768" "800x600" "720x400" "640x480"
    EndSubSection
    SubSection "Display"
        Depth      16
        Modes      "1600x1024" "1280x1024" "1280x960" "1024x768" "800x600" "720x400" "640x480"
    EndSubSection
    SubSection "Display"
        Depth      24
        Modes      "1600x1024" "1280x1024" "1280x960" "1024x768" "800x600" "720x400" "640x480"
    EndSubSection
EndSection

Section "Monitor"
    Identifier  "SGI MultiLink"
    Option      "DPMS"
    HorizSync   28-64
    VertRefresh 43-60
    UseModes    "SGI Multilink Modes"
EndSection

Section "Modes"
    Identifier "SGI Multilink Modes"
    ModeLine "640x480"      25.2  640  656  752  800  480  490  492  525 -Hsync -Vsync
    ModeLine "800x600"      40.0  800  840  968 1056  600  601  605  628 +Hsync +Vsync
    ModeLine "1024x768"     65.0 1024 1048 1184 1344  768  771  777  806 -Hsync -Vsync
    Modeline "1280x960"    108.0 1280 1376 1488 1800  960  961  964 1000 -Hsync -Vsync
    Modeline "1600x1024"   108.0 1600 1616 1656 1704 1024 1027 1030 1056 -Hsync -Vsync
EndSection

```

Hope that helps.  Let me know if you have any questions about how to apply this.

Cheers
