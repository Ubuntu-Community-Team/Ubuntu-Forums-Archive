	# commented out by update-manager, HAL is now used and auto-detects devices

	#    InputDevice    "Keyboard0" "CoreKeyboard"

    InputDevice    "Keyboard0" "CoreKeyboard"
    InputDevice    "Mouse0" "CorePointer"

    # generated from default

    Option         "Emulate3Buttons" "no"
    Option         "ZAxisMapping" "4 5"
EndSection

Section "Monitor"
    Identifier     "Monitor0"
    VendorName     "Unknown"
    ModelName      "Unknown"
    HorizSync       28.0 - 33.0
    VertRefresh     43.0 - 72.0
    Option         "DPMS"
EndSection

Section "Device"
    Identifier     "Device0"
    Driver         "nvidia"
    VendorName     "NVIDIA Corporation"
EndSection

Section "Screen"

    Monitor        "Monitor0"
    DefaultDepth    24
    Option         "NoLogo" "True"
    SubSection     "Display"
        Depth       24
    EndSubSection
EndSection

```
then change the line ```
    Driver         "nvidia"

``` to ```
    Driver         "nv"

```
then reboot 
If that still does not work see if their is a file "xorg.conf.backup" and if their is delete "xorg.conf" and rename "xorg.conf.backup" to "xorg.conf"
