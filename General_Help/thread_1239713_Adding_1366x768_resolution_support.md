### Post by cmost on 2009-08-13
Just a helpful hint...please try Google, or searching these forums before posting questions.  Yours has been asked...and answered at least a zillion times already.

Edit /etc/X11/xorg.conf as root;
Add your desired resolution to the beginning of the "modes" line; before any other resolutions, in quotes (see below.)
Also consider adding the "virtual" line (without the "x") to ensure GDM displays at the desired resolution.

Section "Screen"
    Identifier     "Screen 1"
    Device         "Configured Device"
    Monitor        "Configured Monitor"
    DefaultDepth    24
    SubSection     "Display"
        Depth       24
        Modes      "1400x900" "1024x768" "800x600" "640x480"
        Virtual    1400 900
    EndSubSection
EndSection

### Post by cmost on 2009-08-15
From what I can see from what you posted, you still haven't added the resolution you want to your xorg.conf file.  Here's what mine looks like so you can see the difference.  Ubuntu 9.04+ uses a very minimal xorg.conf file because the new xserver 1.6.x tries to autodetect everything.  Well, obviously it gets it wrong many a time and for people like you, you're stuck with a poor resolution.  Fortunately, anything you add to your xorg.conf file will override xserver's auto-detected settings and you'll end up with the correct resolution.

```
Section "Device"
	Identifier	"Configured Video Device"
	Driver		"nvidia"
#	Driver		"nv"
	Option      	"Coolbits"		"1"
	Option		"NvAGP"			"0"
	Option		"AddARGBGLXVisuals"	"1"
	Option		"AllowGLXWithComposite"	"1"
	Option		"TripleBuffer"		"1"
	Option		"AccelMethod"		"EXA"
	Option		"EXAOptimizeMigration"	"true"
	Option       	"MigrationHeuristic" 	"greedy"
	Option 		"PixmapCacheSize" 	"5000000"	# needed for nvidia 180.xx
#	Option 		"AllowSHMPixmaps" 	"0" 		# needed for nvidia 180.xx
	Option 		"AllowIndirectPixmaps" 	"True"
	Option		"RenderAccel"		"True"
	Option 		"OnDemandVBlankInterrupts"	"True"
	Option 		"InitialPixmapPlacement"	"2"
        Option		"DisableGLXRootClipping"	"True"
	Option		"DamageEvents"		"1"
	Option		"BackingStore"		"1"
	Option 		"UseEvents" 		"False"
EndSection

Section "Monitor"
	Identifier	"Configured Monitor"
EndSection

Section "Screen"
	Identifier	"Default Screen"
	Device		"Configured Video Device"
	Monitor		"Configured Monitor"
	DefaultColorDepth 24
	SubSection "Display"
		Depth	8
		Modes "1280x1024" "1152x864" "1024x768" "800x600" "640x480"
	EndSubSection
	SubSection "Display"
		Depth	16
		Modes "1280x1024" "1152x864" "1024x768" "800x600" "640x480"
	EndSubSection
	SubSection "Display"
		Depth	24
		Modes "1680x1050" "1440x900" "1280x1024" "1152x864" "1024x768" "800x600" "640x480"
		Virtual	1680 1050
	EndSubSection
EndSection

Section "DRI"
	Mode 0666
EndSection

Section "Extensions"
	Option	"Composite"	"Enable"
EndSection
