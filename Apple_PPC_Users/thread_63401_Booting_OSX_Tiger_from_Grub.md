### Post by autocrosser on 2005-09-07
I don't know if anyone on the forum is going to have experience with this one--I run PPC & in the yaboot (mac grub loader) conf is just a simple link to the mac OS & OSX partitions--The mac harddrive needs a "new-world" boot partition, so take a look with gparted to see what it looks like--I know in my system there is--new-world partition-bootstrap partition-Ubuntu partitions & the Mac OSX partition--I'm including my yaboot & fstab for you to look at---hope this helps--

Yaboot first--
## yaboot.conf generated by the Ubuntu installer
##
## run: "man yaboot.conf" for details. Do not make changes until you have!!
## see also: /usr/share/doc/yaboot/examples for example configurations.
##
## For a dual-boot menu, add one or more of:
## bsd=/dev/hdaX, macos=/dev/hdaY, macosx=/dev/hdaZ

boot=/dev/hdb2
message=/boot/boot.msg
device=/pci@f4000000/ata-6@d/disk@1:
partition=5
root=/dev/hdb5
timeout=100
install=/usr/lib/yaboot/yaboot
magicboot=/usr/lib/yaboot/ofboot
fgcolor=light-blue
enablecdboot
macosx=/dev/hda9

image=/boot/vmlinux
    label=Linux
    read-only
    initrd=/boot/initrd.img
    append="quiet splash"

image=/boot/vmlinux.old
    label=old
    read-only
    initrd=/boot/initrd.img.old
    append="quiet splash"

# This entry automatically added by the Ubuntu installer for an existing
# Linux installation on /dev/hda7.
image=/pci@f4000000/ata-6@d/disk@0:7,/boot/vmlinux
    label=hda7-Linux
    root=/pci@f4000000/ata-6@d/disk@0:7
    append="root=/dev/hda7 ro quiet splash"
    initrd=/pci@f4000000/ata-6@d/disk@0:7,/boot/initrd.img

# This entry automatically added by the Ubuntu installer for an existing
# Linux installation on /dev/hda7.
image=/pci@f4000000/ata-6@d/disk@0:7,/boot/vmlinux.old
    label=hda7-old
    root=/pci@f4000000/ata-6@d/disk@0:7
    append="root=/dev/hda7 ro quiet splash"
    initrd=/pci@f4000000/ata-6@d/disk@0:7,/boot/initrd.img.old

Next--fstab--
# /etc/fstab: static file system information.
#
# <file system> <mount point>   <type>  <options>       <dump>  <pass>
proc            /proc           proc    defaults        0       0
/dev/hdb5       /               ext3    defaults,errors=remount-ro 0       1
/dev/hdb6       /home           ext3    defaults        0       2
/dev/hdb10      /opt            ext3    defaults        0       2
/dev/hdb8       /tmp            ext3    defaults        0       2
/dev/hdb7       /usr            ext3    defaults        0       2
/dev/hdb9       /var            ext3    defaults        0       2
/dev/hda7       /oldroot        ext3    defaults        0       2
/dev/hda4       /oldhome        ext3    defaults        0       2
/dev/hda5       /oldopt         ext3    defaults        0       2
/dev/hda3       /oldtmp         ext3    defaults        0       2
/dev/hda9       /mnt/osx        hfsplus defaults        0       0
/dev/hdb4       /mnt/osx3       hfsplus defaults        0       0
/dev/hdc3       /mnt/itunes     hfsplus defaults        0       0
/dev/hdc5       /mnt/games      hfsplus defaults        0       0
/dev/hdd2       /mnt/storage    hfsplus defaults        0       0
/dev/hdb3       none            swap    sw              0       0
/dev/hde        /media/cdrom0   auto    rw,user,noauto  0       0
/dev/hdf        /media/cdrom1   auto    rw,user,noauto  0       0
/dev/scd0       /media/cdrom2   auto    rw,user,noauto  0       0
/dev/sda        /media/usb0     auto    rw,user,noauto  0       0
