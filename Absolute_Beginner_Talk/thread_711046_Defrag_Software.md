### Post by HermanAB on 2008-02-29
Here is a defragger for you:

#! /bin/bash
echo Defragment all disk drives.
echo Copyright H. Acker GPL, 2008
sleep 10
echo Preparing...
sleep 10
df
sleep 20
echo Defragging...
sleep 3600
echo Almost done...
sleep 7200
mount
sleep 1800
echo Getting there...
sleep 600
echo Drum roll...
sleep 10
df
sleep 5
echo Defragment done!
