### Post by alienexplorers on 2007-09-26
Load in the drivers for the video card using ENVY.
Once loaded run:
> gksudo nvidia-settings
and pick out the resolution you want.
You may need to add a modeline and a metamodes line to get the right resolution.
> Section "Monitor"
    Identifier     "Monitor0"
    VendorName     "Unknown"
    ModelName      "Monitor Name"
    HorizSync       your monitors horz sync
    VertRefresh     Your monitors vert refresh
    Option	   "DPMS"
     [COLOR="Red"]# 1440x900 @ 60.00 Hz (GTF) hsync: 55.92 kHz; pclk: 106.47 MHz
     Modeline "1440x900_60.00"  106.47  1440 1520 1672 1904  900 901 904 932  -HSync +Vsync[/COLOR]
EndSection

Section "Device"
    Identifier     "Videocard0"
    Driver         "nvidia"
    VendorName     "NVIDIA Corporation"
    BoardName      "Board Name"
    VideoRam        Memory Amount
EndSection

Section "Screen"
   [COLOR="Red"] Option         "metamodes" "1440x900_60 +0+0"[/COLOR]
    Identifier     "Screen0"
    Device         "Videocard0"
    Monitor        "Monitor0"
    DefaultDepth    24
    Option         "TwinView" "0"
EndSection


Save and reboot
