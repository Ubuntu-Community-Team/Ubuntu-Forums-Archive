I am looking for anyone willing to help me verify something with this kernel on another computer (pqwoerituytrueiwoq: you would be an excellent one to try it, if you see this and are willing). Requirements: An intel processor that will run the intel_pstate driver (some don't). Note: the intel_pstate driver is enabled by default again, whereas for a long time it wasn't. Also, please be aware that it might not go into powersave mode by itself, I think it does for Utopic, but doesn't for Trusty (which is what I am running). [See also]("https://bugs.launchpad.net/ubuntu/+source/sysvinit/+bug/1314653") or just do it manually.

Example to check which driver is running:```

[COLOR=#ff0000]for file in /sys/devices/system/cpu/cpu*/cpufreq/scaling_governor; do echo "powersave" > $file; done[/COLOR]

```Now, what I am asking to be done: First, compile this program (consume.c):```
/******************************************************
/*
/* consume ratio freq
/*         to consume cpu time at ratio and frequency.
/*         The original code was from Peter Zijlstra,
/*         the kernal.org maintainer of the sched
/*         area of the kernel code.
/*
/* consume.c Smythies 2014.06.07
/*         Fix dense math issue with elapsed time.
/*
/* consume.c Smythies 2014.05.31
/*         Even though I don't like the extra overhead,
/*         List its PID and some times to assist
/*         correlating with "perf record" data.
/*
/* consume.c Smythies 2014.05.05
/*         Fix usage message.
/*
/* consume.c Smythies 2013.11.04
/*         Add a time to execute parameter.
/*         allow the fraction to be as low as 0.5 percent
/*
/* consume.c Smythies 2012.06.27
/*         More precision desired. Switch to doubles for
/*         run time parameters.
/*
/* consume.c Smythies 2012.06.23
/*         I think in frequency for this stuff, so change
/*         the run time parameter from period to frequency.
/*         Make it not optional.
/*
/* consume.c Smythies 2012.06.23
/*         First just try to compile the code as sent
/*         from Peter Zijlstra.
/*         But converted to my coding standards (which will
/*         annoy Peter if it ever gets sent back).
/*
/******************************************************/

#include <sys/time.h>
#include <sys/types.h>
#include <unistd.h>
#include <stdio.h>
#include <stdlib.h>

unsigned long long stamp(void){
   struct timeval tv;

   gettimeofday(&tv, NULL);

   return (unsigned long long)tv.tv_sec * 1000000 + tv.tv_usec;
} /* endprocedure */

void consume(int spin, int total, int elapsed){
   unsigned long long begin, now, start, lelapsed;

   begin = stamp();
   start = begin;
   now = begin;
   lelapsed = (unsigned long long)elapsed * (unsigned long long) 1000000;

   while((long long)(now - start) < lelapsed) {  /* for the requested time */
      now = stamp();
      if ((long long)(now - begin) > spin) {
         usleep(total - spin);
         begin += total;
      } /* endif */
   } /* endwhile */
   printf("  Elapsed: %llu  Now: %llu\n", (now - start), now);
} /* endprocedure */

int main(int argc, char **argv){
   double frac, freq, duration, period;

   printf("consume: %s %s %s  PID: %d", argv[1], argv[2], argv[3], getpid());
   switch(argc){
   case 4:
      frac = atof(argv[1]);
      freq = atof(argv[2]);
      duration = atof(argv[3]);
      break;
   default:
      fprintf(stderr, "%s <frac> <freq> <duration>\n  frac   -- [0.5-100.0] %% of time to burn\n  freq -- [hz] (min 1.0) frequency of burn/sleep cycle\n duration -- [seconds] (0.0 means 1 day) time for program to run\n", argv[0]);
      return -1;
   } /* endcase */

/*   printf("consume: %lf %lf %lf\n", frac, freq, duration); /*  */
   if (frac > 100.0){
      frac = 100.0;
   } /* endif */
   if (frac < 0.50){
      frac = 0.50;
   } /* endif */
   if (freq < 1.0){
      freq = 1.0;
   } /* endif */
   if (duration <= 0.0){
      duration = 86400.0;
   } /* endif */
   period = 1000000.0 / freq;
/*   printf("consume: %lf %lf %lf %lf\n", frac, freq, period, duration); /*  */
   consume((int)(period * frac / 100.0 + 0.5), (int)(period + 0.5), (int)(duration + 0.5));
   return 0;
} /* endprocedure */

```How to compile (from the same directory as the source code):```
cc consume.c -o consume
```Then run this script several times (if you are running the low latency kernel then uncomment the appropriate line and comment out the 250 Hz one. If your processor has less CPUs, adjust things accordingly. Adjust paths as per your system):```
doug@s15:~/temp$ [COLOR=#ff0000]cat set_cpu_timer_stats[/COLOR]
#! /bin/bash
# Reset the counters
echo "0" > /proc/timer_stats
echo "1" > /proc/timer_stats
# For a 1000 Hz kernel and for 40% probability of issue:
# Apply a 6 percent load at 100 hertz for 20 seconds
# taskset -c 5 /home/doug/c/consume 6.0 100.0 20.0 &
# For a 250 Hz kernel and for 40% probability of issue:
# Apply a 30 percent load at 125 hertz for 20 seconds
taskset -c 5 /home/doug/c/consume 30.0 125.0 20.0 &
sleep 10
# What is the CPU 5 frequency?
echo -n "CPU 5 frequency (~ 1.6 GHz means forced low, ~~higher means O.K.): "
cat /sys/devices/system/cpu/cpu5/cpufreq/cpuinfo_cur_freq
sleep 11
cat /proc/timer_stats | grep intel
```Note the low frequency part of the message is for my computer, it would be much lower for a haswell processor, for example.
