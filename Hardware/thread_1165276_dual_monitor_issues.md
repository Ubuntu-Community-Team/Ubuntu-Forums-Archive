### Post by hmsims on 2009-06-17
[COLOR=Blue]Here is what the final configuration file looked like:

[COLOR=Black]Section "InputDevice"
    Identifier    "Generic Keyboard"
    Driver        "kbd"
    Option        "XkbRules"    "xorg"
    Option        "XkbModel"    "pc105"
    Option        "XkbLayout"    "us"
EndSection

Section "InputDevice"
    Identifier    "Configured Mouse"
    Driver        "mouse"
EndSection

Section "Device"
    Identifier    "Configured Video Device"
    Boardname    "NVIDIA GeForce"
    Busid        "PCI:1:0:0"
    Driver        "nvidia"
    Vendorname    "NVIDIA"
    Option        "RenderAccel"    "True"
    Option        "TwinView"    "1"
    Option        "MetaModes"    "1280x1024"
    Option        "NoLogo"    "True"
EndSection

Section "Monitor"
    Identifier    "Configured Monitor"
    Vendorname    "Plug 'n' Play"
EndSection

Section "Screen"
    Identifier    "Default Screen"
    Device        "Configured Video Device"
    Monitor        "Configured Monitor"
    Option        "TwinView"    "1"
    Option        "MetaModes"    "1280x1024 1280x1024"
    Option        "TwinViewOrientation"    "RightOf"
EndSection


Section "ServerLayout"
    Identifier    "Default Layout"
  screen 0 "Default Screen" 0 0
    Option        "Xinerama"    "Off"
EndSection

Section "Module"
    Load        "glx"
    Load        "v4l"
EndSection

Section "ServerFlags"
EndSection
