### Post by 3shmaoy on 2008-10-23
I had a similar problem with a the same graphics card but exactly the same.
May be this part of the xorg.conf can help you.
I got most of it from that link
[http://www.intellinuxgraphics.org/dualhead.html]("http://www.intellinuxgraphics.org/dualhead.html")
But I had to fix a few lines.
I hope this can help.

```

Section "Device"
        Identifier      "Intel Corporation Mobile GM965/GL960 Integrated Graphics Controller rev 3"
        Boardname       "Intel 965"
        Busid           "PCI:0:2:0"     # you can get the exact values from Xorg.log
        Driver          "intel"
        Option          "monitor-VGA"           "VGA Monitor"
        Option          "monitor-LVDS"          "laptop LCD"
        Vendorname      "Intel"
EndSection

Section "Monitor"
        Identifier      "VGA Monitor"
        Option          "PreferredMode"         "1280x1024"
        Option          "Position"              "0 0"
        Option          "Enabled"               "true"
EndSection


Section "Monitor"
        Identifier      "laptop LCD"
        Option          "PreferredMode"         "1024x768"
        Option          "RightOf"               "VGA Monitor"
        Option          "Enabled"               "true"
EndSection

Section "Screen"
        Identifier      "Default Screen"
        Device          "Intel Corporation Mobile GM965/GL960 Integrated Graphics Controller rev 3"
        Monitor         "VGA Monitor"
        SubSection "Display"
                        Virtual                 2304 1024 # should be equal to a rectangle covering all monitors
                        Modes                   "1024x768@60" "1280x1024@60" # required modes & freq.s of the monitors
        EndSubSection
EndSection

Section "ServerLayout"
        Identifier      "Default Layout"
        screen          "Default Screen"
        Inputdevice     "Synaptics Touchpad"
EndSection

Section "Module"
        Load            "glx"
        Load            "GLcore"
        Load            "v4l"
EndSection

Section "ServerFlags"
EndSection


```
