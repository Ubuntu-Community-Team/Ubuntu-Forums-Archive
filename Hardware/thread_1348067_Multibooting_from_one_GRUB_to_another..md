### Post by oldfred on 2009-12-07
You normally need grub installed to a partition to chainboot but you can also chainboot to the MBR of another drive. if it is drive 1 (second drive or sdb), usual entry is drive,partition like (hd0,0)

grub legacy at end of menu.lst
title      Chainload mbr hd1
root        (hd1)
chainloader +1

grub2 add to 40_custom and run update-grub
# Chainload another bootloader
menuentry "Chainload mbr hd1" {
set root=(hd1)
chainloader +1
}

### Post by oldfred on 2009-12-08
Your entries look awfully sparse. I have seen where you can eliminate lines to make it work but these are my entries. Of course you need your UUID and drive letters. If window is not on the first drive you will need the drivemap command with (hd1) or (hd2) rather than (hd0). My entry should not even be required since is is mapping hd0 to root which should also be 0.

### BEGIN /etc/grub.d/30_os-prober ###
menuentry "Microsoft Windows XP Professional (on /dev/sda1)" {
    insmod ntfs
    set root=(hd0,1)
    search --no-floppy --fs-uuid --set 04b05b70b05b6768
    drivemap -s (hd0) ${root}
    chainloader +1
}
menuentry "Ubuntu 9.04, kernel 2.6.28-17-generic (on /dev/sdb1)" {
    insmod ext2
    set root=(hd1,1)
    search --no-floppy --fs-uuid --set f9b2a783-78ba-4437-9bc5-9d8cfa86ff0c
    linux /boot/vmlinuz-2.6.28-17-generic root=UUID=f9b2a783-78ba-4437-9bc5-9d8cfa86ff0c ro splash quiet
    initrd /boot/initrd.img-2.6.28-17-generic
}
