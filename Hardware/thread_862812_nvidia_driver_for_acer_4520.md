### Post by Quitty on 2009-02-20
If i were you i would uninstall (nvidia.run) and reinstall form 
System > Admin > Hardware Drivers
 
I also have the Acer 4520 Hardy 32bit. Heres a copy of my /etc/X11/xorg.conf

It's set up for dual monitors and svideo out works for me.


# xorg.conf (X.Org X Window System server configuration file)

# This file is automatically updated on xserver-xorg package upgrades *only*

# package.

Section "InputDevice"
	Identifier	"Generic Keyboard"
	Driver		"kbd"
	Option		"XkbRules"	"xorg"
	Option		"XkbModel"	"pc105"
	Option		"XkbLayout"	"us"
EndSection

Section "InputDevice"
	Identifier	"Configured Mouse"
	Driver		"mouse"
	Option		"CorePointer"
EndSection

Section "InputDevice"
	Identifier	"Synaptics Touchpad"
	Driver		"synaptics"
	Option		"SendCoreEvents"	"true"
	Option		"Device"	"/dev/psaux"
	Option		"Protocol"	"auto-dev"
	Option		"HorizEdgeScroll"	"0"
EndSection

Section "Device"
	Identifier	"Configured Video Device"
	Driver		"nvidia"
	Option          "RenderAccel"   "true"
        Option		"NoLogo"	"True"
EndSection

Section "Monitor"
	Identifier	"Configured Monitor"
EndSection

Section "Screen"
	Identifier	"Default Screen"
	Monitor		"Configured Monitor"
	Device		"Configured Video Device"
	Defaultdepth	24
##This turns on NVidia’s TwinView
        Option "TwinView"
        Option  "TwinViewOrientation" "Clone"
        Option "TVOutFormat" "S-VIDEO"
        Option "TVStandard" "NTSC-M"
        Option "SecondMonitorHorizSync" "30-50"
        Option "SecondMonitorVertRefresh" "60"
        Option "MetaModes" "1024x768,1024x768;800x600,800x600;640x480,640x480; 512x384,512x384"
EndSection

Section "ServerLayout"
	Identifier	"Default Layout"
  screen "Default Screen"
	Inputdevice	"Synaptics Touchpad"
EndSection
Section "Module"
	Load		"glx"
EndSection

Good luck.
