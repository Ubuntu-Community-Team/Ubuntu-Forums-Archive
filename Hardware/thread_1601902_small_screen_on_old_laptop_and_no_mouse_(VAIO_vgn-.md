### Post by pixel :-) on 2010-11-15
here's the fix for the screen, you need to edit /etc/X11/xorg.conf
> # nvidia-xconfig: X configuration file generated by nvidia-xconfig
# nvidia-xconfig:  version 1.0  (buildmeister@builder63)  Wed Sep 29 10:19:47 PDT 2010

Section "Monitor"
       Identifier   "Monitor0"
       VendorName   "Sony"
       ModelName    "X-Black"
       Option       "dpms"
       HorizSync    29-60
       VertRefresh  0-60
       DisplaySize  285 179 #important to set dpi properly
       Modeline "1280x800"  68.56  1280 1336 1472 1664  800 801 804 824  -HSync +Vsync
EndSection


Section "Screen"
	Identifier     "Screen0"
	Device         "Device0"
	Monitor        "Monitor0"
	DefaultDepth	24
	SubSection "Display"
		Modes	"1280x800"
		Depth       24
	EndSubSection
EndSection

Section "Module"
	Load	"glx"
EndSection

	Identifier     "Keyboard0"
	Driver         "kbd"
	# generated from default
EndSection

Section "ServerLayout"
	Identifier     "Layout0"
	Screen      0  "Screen0"
	InputDevice    "Keyboard0" "CoreKeyboard"
EndSection

Section "Device"
	Identifier     "Device0"
	VendorName     "NVIDIA Corporation"
	Option	"NvAGP" "0"
	Driver	"nvidia"
	Option	"NoLogo"	"False"
	Option	"ExactModeTimingsDVI" "True"
	Option	"ModeValidation" "NoHorizSyncCheck,NoDFPNAtiveResolutionCheck,NoExtendedGpuCapabilitiesCheck,NoWidthAlignmentCheck,NoVertRefreshCheck"
EndSection
