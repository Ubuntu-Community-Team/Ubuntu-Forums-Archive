### Post by henrik_daver on 2009-03-03
After reinstalling Ubuntu it thinks I have two monitors on my laptop. I have a Clevo laptop ([http://www.clevo.com.tw/en/products/prodinfo_2.asp?productid=88](http://www.clevo.com.tw/en/products/prodinfo_2.asp?productid=88)) with Intel video controller, which worked perfectly with the intel driver before the boot. Unfortunately I wasn't foreseeing enough to backup the xorg.conf before reformatting...

I can set the screen resolution through xorg.conf,

```
Section "Device"
	Identifier	"Configured Video Device"
	Driver		"intel"
EndSection

Section "Monitor"
	Identifier	"Configured Monitor"
EndSection

Section "Screen"
	Identifier	"Default Screen"
	Monitor		"Configured Monitor"
	Device		"Configured Video Device"
	SubSection	"Display"
		Depth	24
		Modes	"1440x900" "1024x768"
	EndSubSection
EndSection
```

but it only seems to affect one of the monitors; the other one I don't know how to change the resolution for.

In xorg.conf I've also tried to set the driver to "intel" manually, but as the lshw output below shows, it doesn't seem to make any difference:


So:
Is there any non-xorg.conf way for me to "remove" the extra monitor?

Grateful for help,

-------------------------------------------
At last I gave up and mailed my computer vendor, who sent me this xorg.conf file, and now everything works fine.

```
# xorg.conf (xorg X Window System server configuration file)
#
# This file was generated by dexconf, the Debian X Configuration tool, using
# values from the debconf database.
#
# Edit this file with caution, and see the xorg.conf manual page.
# (Type "man xorg.conf" at the shell prompt.)
#
# This file is automatically updated on xserver-xorg package upgrades *only*
# if it has not been modified since the last upgrade of the xserver-xorg
# package.
#
# If you have edited this file but would like it to be automatically updated
# again, run the following command:
#   sudo dpkg-reconfigure -phigh xserver-xorg

Section "Files"
EndSection

Section "InputDevice"
	Identifier	"Generic Keyboard"
	Driver		"kbd"
	Option		"CoreKeyboard"
	Option		"XkbRules"	"xorg"
	Option		"XkbModel"	"pc105"
	Option		"XkbLayout"	"se"
	Option		"XkbOptions"	"lv3:ralt_switch"
EndSection

Section "InputDevice"
        Identifier      "Configured Mouse"
        Driver          "mouse"
        Option          "CorePointer"
EndSection

Section "InputDevice"
	Identifier	"Synaptics Touchpad"
	Driver		"synaptics"
	Option		"SendCoreEvents"	"true"
	Option		"Device"		"/dev/psaux"
	Option		"Protocol"		"auto-dev"
	Option		"HorizScrollDelta"	"0"
	Option		"SHMConfig"		"True"
EndSection

Section "Device"
	Identifier	"Intel Corporation Cantiga Integrated Graphics Controller [GGS-Data]"
	Driver		"intel"
	Option		"monitor-VGA"		"VGAOutput"
	Option		"monitor-LVDS"		"LVDSOutput"
        Option          "monitor-HDMI-1"        "HDMIOutput1"
	Option		"monitor-HDMI-2"	"HDMIOutput2"
#	Option		"NoXVideo"
EndSection

Section "Monitor"
	Identifier	"LVDSOutput"
	Option		"DPMS"
EndSection

Section "Monitor"
        Identifier      "HDMIOutput1"
EndSection

Section "Monitor"
        Identifier      "HDMIOutput2"
        Option          "Ignore"        "true"
EndSection

Section "Monitor"
        Identifier      "VGAOutput"
EndSection

Section "Screen"
	Identifier	"Default Screen"
	Device		"Intel Corporation Cantiga Integrated Graphics Controller [GGS-Data]"
	Monitor		"LVDSOutput"
	DefaultDepth	24
	SubSection "Display"
#		Virtual		2720	1024
	EndSubSection
EndSection

Section "Screen"
        Identifier      "HDMI Screen"
        Device          "Intel Corporation Cantiga Integrated Graphics Controller [GGS-Data]"
        Monitor         "HDMIOutput1"
        DefaultDepth 24
        Subsection "Display"
        EndSubsection
EndSection

Section "Screen"
        Identifier      "VGA Screen"
	Device		"Intel Corporation Cantiga Integrated Graphics Controller [GGS-Data]"
        Monitor         "VGAOutput"
        DefaultDepth 24
        Subsection "Display"
        EndSubsection
EndSection

Section "ServerLayout"
	Identifier	"Default Layout"
	Screen		"Default Screen"
#	Screen          "Default Screen" 0 0
#	Screen          "VGA Screen" RightOf "Default Screen"
	InputDevice	"Generic Keyboard"
	InputDevice	"Configured Mouse"
	InputDevice	"Synaptics Touchpad"
EndSection
```

Apparently, the computer has four instances that are seen as "monitors", and disabling one with the "Ignore" "true" setting made it work.
