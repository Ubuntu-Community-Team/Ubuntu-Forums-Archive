### Post by fozters on 2013-11-03
OK, I got it sorted out..

This was the right way to go:

KERNEL=="sd?", SUBSYSTEM=="block", DEVPATH=="*usb1/1-4*", NAME="sdu",  RUN+="/usr/bin/logger My disk ATTR{partition}=$ATTR{partition},  DEVPATH=$devpath, ID_PATH=$ENV{ID_PATH}, ID_SERIAL=$ENV{ID_SERIAL}",  GOTO="END_20_PERSISTENT_DISK" 
LABEL="END_20_PERSISTENT_DISK"

But kernel won't accept hard links(from syslog):
Nov  3 14:04:09 lamp udevd[716]: kernel-provided name 'sdg' and NAME= 'sdu' disagree, please use SYMLINK+= or change the kernel to provide the proper name

So here is how is setup the rules file:


root@lamp:/home/fozters# more /etc/udev/rules.d/20-disk-bays.rules
#FIRST ROOT DISK (ORIGINAL SDE-1/3) SYMLINKED TO ROOTA-1/3
KERNEL=="sd?", SUBSYSTEM=="block", DEVPATH=="*usb1/1-1*", SYMLINK+="roota", RUN+="/usr/bin/logger My disk ATTR{partition}=$ATTR{partition}, DEVPATH=$devpath, ID_PATH=$ENV{ID_PATH
}, ID_SERIAL=$ENV{ID_SERIAL}", GOTO="END_20_PERSISTENT_DISK"
KERNEL=="sd?*", ATTR{partition}=="1", SUBSYSTEM=="block", DEVPATH=="*usb1/1-1*", SYMLINK+="roota%n" RUN+="/usr/bin/logger My partition parent=%p number=%n, ATTR{partition}=$ATTR{
partition}"
KERNEL=="sd?*", ATTR{partition}=="2", SUBSYSTEM=="block", DEVPATH=="*usb1/1-1*", SYMLINK+="roota%n" RUN+="/usr/bin/logger My partition parent=%p number=%n, ATTR{partition}=$ATTR{
partition}"
KERNEL=="sd?*", ATTR{partition}=="3", SUBSYSTEM=="block", DEVPATH=="*usb1/1-1*", SYMLINK+="roota%n" RUN+="/usr/bin/logger My partition parent=%p number=%n, ATTR{partition}=$ATTR{
partition}"

#SECOND ROOT DISK (ORIGINAL SDF-1/3) SYMLINKED TO ROOTB-1/3
KERNEL=="sd?", SUBSYSTEM=="block", DEVPATH=="*usb1/1-2*", SYMLINK+="rootb", RUN+="/usr/bin/logger My disk ATTR{partition}=$ATTR{partition}, DEVPATH=$devpath, ID_PATH=$ENV{ID_PATH
}, ID_SERIAL=$ENV{ID_SERIAL}", GOTO="END_20_PERSISTENT_DISK"
KERNEL=="sd?*", ATTR{partition}=="1", SUBSYSTEM=="block", DEVPATH=="*usb1/1-2*", SYMLINK+="rootb%n" RUN+="/usr/bin/logger My partition parent=%p number=%n, ATTR{partition}=$ATTR{
partition}"
KERNEL=="sd?*", ATTR{partition}=="2", SUBSYSTEM=="block", DEVPATH=="*usb1/1-2*", SYMLINK+="rootb%n" RUN+="/usr/bin/logger My partition parent=%p number=%n, ATTR{partition}=$ATTR{
partition}"
KERNEL=="sd?*", ATTR{partition}=="3", SUBSYSTEM=="block", DEVPATH=="*usb1/1-2*", SYMLINK+="rootb%n" RUN+="/usr/bin/logger My partition parent=%p number=%n, ATTR{partition}=$ATTR{
partition}"

#FIRST DATA DISK (ORIGINAL SDA-1) SYMLINKED TO DATAA-1
KERNEL=="sd?", SUBSYSTEM=="block", DEVPATH=="*ata1/host0*", SYMLINK+="dataa", RUN+="/usr/bin/logger My disk ATTR{partition}=$ATTR{partition}, DEVPATH=$devpath, ID_PATH=$ENV{ID_PA
TH}, ID_SERIAL=$ENV{ID_SERIAL}", GOTO="END_20_PERSISTENT_DISK"
KERNEL=="sd?*", ATTR{partition}=="1", SUBSYSTEM=="block", DEVPATH=="*ata1/host0*", SYMLINK+="dataa%n" RUN+="/usr/bin/logger My partition parent=%p number=%n, ATTR{partition}=$ATT
R{partition}"

#SECOND DATA DISK (ORIGINAL SDB-1) SYMLINKED TO DATAB-1
KERNEL=="sd?", SUBSYSTEM=="block", DEVPATH=="*ata2/host1*", SYMLINK+="datab", RUN+="/usr/bin/logger My disk ATTR{partition}=$ATTR{partition}, DEVPATH=$devpath, ID_PATH=$ENV{ID_PA
TH}, ID_SERIAL=$ENV{ID_SERIAL}", GOTO="END_20_PERSISTENT_DISK"
KERNEL=="sd?*", ATTR{partition}=="1", SUBSYSTEM=="block", DEVPATH=="*ata2/host1*", SYMLINK+="datab%n" RUN+="/usr/bin/logger My partition parent=%p number=%n, ATTR{partition}=$ATT
R{partition}"

#THIRD DATA DISK (ORIGINAL SDC-1) SYMLINKED TO DATAC-1
KERNEL=="sd?", SUBSYSTEM=="block", DEVPATH=="*ata3/host2*", SYMLINK+="datac", RUN+="/usr/bin/logger My disk ATTR{partition}=$ATTR{partition}, DEVPATH=$devpath, ID_PATH=$ENV{ID_PA
TH}, ID_SERIAL=$ENV{ID_SERIAL}", GOTO="END_20_PERSISTENT_DISK"
KERNEL=="sd?*", ATTR{partition}=="1", SUBSYSTEM=="block", DEVPATH=="*ata3/host2*", SYMLINK+="datac%n" RUN+="/usr/bin/logger My partition parent=%p number=%n, ATTR{partition}=$ATT
