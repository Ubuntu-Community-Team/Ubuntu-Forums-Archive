### Post by tomchuk on 2007-03-22
You can use two methods to do this with the free radeon driver. The first, which I've had much more luck with in the past is Xinerama. This has been the de-facto way to do multi-monitor setups on X for years. It's simple and proven, but you lose hardware 3D acceleration. The second is Merged Framebuffer - a single large framebuffer with two viewports looking in to it. I've never been able to get this to work on my laptop, but I know people have. I haven't tried in a while though, so let me post my working xorg.conf for Xinerama, then try to get MergedFD working and I'll give you an update with a working config if I can. It seems the new mode setting stuff in Xorg 7.2 (feisty) should make this easier.

xorg.conf
```

  VertRefresh 60
```

The most important thing to note are the two Device sections each with a screen property and the two monitor sections. Also make sure you have the screen lines from the ServerLayout section and the Xinerama properties from ServerLayout and ServerFlags. Also note that this is for a 1400x1050 main screen and a 1600x1200 secondary screen to the right - adjust these values and position to your configuration.

### Post by tomchuk on 2007-03-22
Wow, that was easy. Got MergedFB going on the first try. I've tried to get this config working dozens of times on xorg 7.1 and 7.0 to no avail. It seems the new modesetting stuff in 7.2 is really what did the trick. As before this is a setup for 1400x1050 on the main screen, 1600x1200 on the second. The important things to note are the lines in the device section after the empty line, and you must include any resolutions used in the screen section under modes. In MetaModes you can list 2 resolutions for each monitor separated by a hyphen for streched mode, or a single resolution for clone mode (good for presentations). This setup allows me to use the max resolutions of my monitors in stretch mode, or clone mode in a variety of resolutions in clone mode.

```

#
# This file was generated by dexconf, the Debian X Configuration tool, using
# values from the debconf database.

# (Type "man /etc/X11/xorg.conf" at the shell prompt.)
#
# This file is automatically updated on xserver-xorg package upgrades *only*
# if it has not been modified since the last upgrade of the xserver-xorg
# package.
#
# If you have edited this file but would like it to be automatically updated
# again, run the following command:
#   sudo dpkg-reconfigure -phigh xserver-xorg

Section "Files"
    FontPath    "/usr/share/fonts/X11/misc"
    FontPath    "/usr/share/fonts/X11/cyrillic"
    FontPath    "/usr/share/fonts/X11/100dpi/:unscaled"
    FontPath    "/usr/share/fonts/X11/75dpi/:unscaled"
    FontPath    "/usr/share/fonts/X11/Type1"
    FontPath    "/usr/share/fonts/X11/100dpi"
    FontPath    "/usr/share/fonts/X11/75dpi"

    Load    "ddc"
    Load    "dri"
    Load    "extmod"
    Load    "freetype"
    Load    "glx"
    Load    "int10"
    Load    "vbe"

    Identifier  "KB"
    Driver      "kbd"
    Option      "CoreKeyboard"
    Option      "XkbRules"  "xorg"
    Option      "XkbModel"  "pc101"
    Option      "XkbLayout" "us"
    Option      "XkbOptions"    "ctrl:nocaps"

Section "InputDevice"
    Identifier  "MOUSE"
    Driver      "mouse"
    Option      "CorePointer"
    Option      "Device"        "/dev/input/mice"
    Option      "Protocol"      "ImPS/2"
    Option      "Emulate3Buttons"   "true"

Section "Device"
    Identifier  "ATI"

    Option      "AccelMethod"       "XAA"
    Option      "EnablePageFlip"    "true"
    Option      "BIOSHotkeys"       "on"
    Option      "DynamicClocks"     "on"
    Option      "DRI"                   "true"
    Option      "XAANoOffscreenPixmaps" "true"
    Option      "AddARGBGLXVisuals"     "true"
    Option      "AllowGLXWithComposite" "true"

    Option      "MergedFB"          "true"
    Option      "CRT2Position"      "RightOf"
    Option      "CRT2HSync"         "30-80"
    Option      "CRT2VRefresh"      "60"
    Option      "MetaModes"         "1400x1050-1600x1200 1280x960 1400x1050 1024x768 800x600"
    Option      "MergedNonRectangular" "true"
    Option      "MergedDPI"         "100 100"
    Option      "MergedXinerama"    "true"
EndSection

Section "Monitor"
    Identifier  "MON"
    Option      "DPMS"
    HorizSync   28-70
    VertRefresh 43-72
EndSection

Section "Screen"
    Identifier  "SCR"
    Device      "ATI"
    Monitor     "MON"
    DefaultDepth    24
    SubSection "Display"
    #   Virtual 3000 1200
        Depth       24
        Modes       "1600x1200" "1400x1050" "1280x960" "1024x768" "800x600"
    EndSubSection
EndSection

Section "ServerLayout"
    Identifier  "LAY"
    Screen      "SCR"
    InputDevice "KB"
    InputDevice "MOUSE"
EndSection

Section "Extensions"
    Option      "Composite"          "Enable"
EndSection

Section "ServerFlags"
    Option      "OffTime"       "1"
EndSection

Section "DRI"
    Mode    0666
EndSection

```
