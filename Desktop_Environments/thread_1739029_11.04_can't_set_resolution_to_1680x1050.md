### Post by jpmiranda on 2011-07-10
Hi,

Try this xorg.conf configuration. It was OK for me.

******************************

# generate modeline with command "cvt 1680 1050" (or the resolution you're looking for)

Section "Device"
Identifier "Configured Video Device"
# Option "ModeValidation" "NoEdidModes"
# Option   "UseEdid" "false"
EndSection

Section "Monitor"
Identifier "Configured Monitor"
Vendorname "Generic LCD Display"
Modelname "LCD Panel"
Horizsync 30.0-81.0
Vertrefresh 56.0 - 75.0
modeline "1024x768_60.00" 65.0 1024 1048 1184 1344 768 771 777 806 -vsync -hsync
modeline "1440x900_60.00" 106.47 1440 1520 1672 1904 900 901 904 932 -HSync +Vsync
modeline "1680x1050_60.00"  146.25  1680 1784 1960 2240  1050 1053 1059 1089 -hsync +vsync
Gamma 1.0
# modeline "640x480@60" 25.2 640 656 752 800 480 490 492 525 -vsync -hsync
# modeline "800x600@56" 36.0 800 824 896 1024 600 601 603 625 +hsync +vsync
# modeline "800x600@60" 40.0 800 840 968 1056 600 601 605 628 +hsync +vsync
# modeline "1024x768@60" 65.0 1024 1048 1184 1344 768 771 777 806 -vsync -hsync
# modeline "1440x900_60.00" 106.47 1440 1520 1672 1904 900 901 904 932 -HSync +Vsync
# modeline "1680x1050_60.00"  146.25  1680 1784 1960 2240  1050 1053 1059 1089 -hsync +vsync
EndSection

Section "Screen"
Identifier "Default Screen"
Monitor "Configured Monitor"
Device "Configured Video Device"
DefaultDepth 24
SubSection "Display"
Depth 1
Modes "1680x1050" "1440x900" "1024x768"
EndSubSection
SubSection "Display"
Depth 4
Modes "1680x1050" "1440x900" "1024x768"
EndSubSection
SubSection "Display"
Depth 8
Modes "1680x1050" "1440x900" "1024x768"
EndSubSection
SubSection "Display"
Depth 15
Modes "1680x1050" "1440x900" "1024x768"
EndSubSection
SubSection "Display"
Depth 16
Modes "1680x1050" "1440x900" "1024x768"
EndSubSection
SubSection "Display"
Depth 24
Modes "1680x1050" "1440x900" "1024x768"
EndSubSection
EndSection

*********************************

Good luck!

### Post by rosencrantz on 2011-07-10
A colleague of mine had the same problem: he had to set an explicit modeline in xorg.conf which made the mode accessible to xrandr.
The modeline parameters depend on your screen, so either find an example online or calculate the values yourself.
[http://howto-pages.org/ModeLines/](http://howto-pages.org/ModeLines/)
[http://www.arachnoid.com/modelines/index.html](http://www.arachnoid.com/modelines/index.html)
[http://www.mcamafia.de/mcapage0/timecalc.htm](http://www.mcamafia.de/mcapage0/timecalc.htm)
