### Post by stueyboy on 2007-02-05
HI. I have a 9600pro and have tried a few times to get it working with xgl with no success. It does however work fine with AIGLX and looks pretty good with it. I followed various installation howtos on here to get it going and the end result of the xorg.conf looks like this if you are interested in trying.

> Section "Device"
	Identifier	"ATI Technologies, Inc. RV350 AP [Radeon 9600]"
	Driver		"radeon"
	Option		"AGPFastWrite" "yes"
	Option		"AGPMode" "8"
	Option		"ColorTiling" "on"
	Option		"EnablePageFlip" "true"
	Option		"AccelMethod" "EXA"
	Option		"XAANoOffScreenPixMaps"
	Option		"RenderAccel" "true"
	Option		"DRI" "true"
	BusID		"PCI:1:0:0"
EndSection

> Section "ServerLayout"	
	Option		"AIGLX"	"True"			
	Identifier	"Default Layout"
	Screen		"Default Screen"
	InputDevice	"Generic Keyboard"
	InputDevice	"Configured Mouse"
	InputDevice     "stylus" "SendCoreEvents"
	InputDevice     "cursor" "SendCoreEvents"
	InputDevice     "eraser" "SendCoreEvents"
EndSection

Section "DRI"
	Mode	0666
EndSection

Section	"Extensions"
	Option	"composite" "Enable"
EndSection
