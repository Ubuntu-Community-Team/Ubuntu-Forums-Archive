Here is my working xorg.conf for my ATI Radeon 9550 for your reference, I guess it would work for you even if you were to use it entirely.

```

Section "Device"
	Identifier	"Sapphire ATI Radeon 9550 - Analog Output"
	Driver		"fglrx"
	Screen		0
	BusID		"PCI:1:0:0"
EndSection

Section "Device"
	Identifier      "Sapphire ATI Radeon 9550 - DVI Output"
	Driver          "fglrx"
	Screen		1
	BusID           "PCI:1:0:0"
EndSection

Section "Monitor"
	Identifier	"Philps 107S6"
	Option		"DPMS"
	HorizSync	30-71
	VertRefresh	50-160
EndSection

Section "Monitor"
	Identifier      "Hewlett Packard A4575A"
	Option          "DPMS"
	HorizSync       20-100
	VertRefresh     50-120
EndSection

Section "Screen"
	Identifier	"Analog Output"
	Device		"Sapphire ATI Radeon 9550 - Analog Output"
	Monitor		"Philps 107S6"
	DefaultDepth	24
	SubSection "Display"
		Depth		24
		Modes		"1024x768" "800x600" "640x480"
	EndSubSection
EndSection

Section "Screen"
	Identifier	"DVI Output"
	Device		"Sapphire ATI Radeon 9550 - DVI Output"
	Monitor		"Hewlett Packard A4575A"
	DefaultDepth	24
	SubSection "Display"
		Depth		24
		Modes		"1024x768" "800x600" "640x480"
	EndSubSection
EndSection

Section "ServerLayout"
	Identifier	"Default Layout"
	Screen	0	"Analog Output"
	Screen	1	"DVI Output"	RightOf	"Analog Output"
	InputDevice	"Generic Keyboard"
	InputDevice	"Configured Mouse"
EndSection
