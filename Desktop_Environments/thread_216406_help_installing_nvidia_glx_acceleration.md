### Post by Perfect Storm on 2006-07-15
Change:

Section "Device"
	Identifier	"NVIDIA Corporation NV43 [GeForce 6600/GeForce 6600 GT]"
	Driver		"vesa"
	BusID		"PCI:1:0:0"

to


ection "Device"
	Identifier	"NVIDIA Corporation NV43 [GeForce 6600/GeForce 6600 GT]"
	Driver		"nvidia"
	BusID		"PCI:1:0:0"





restart
