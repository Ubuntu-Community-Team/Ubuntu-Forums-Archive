### Post by mojoman on 2008-03-05
In the Section "ServerLayout" try adding

```
        Option         "Xinerama" "true"
```

If that doesn't do it, try it as false. I think this is the option which allows for two screens, i.e. not having applications spanning both screens. And of course you can freely move windows between the screens. 

/mojoman

### Post by citybird on 2008-03-05
I just installed 64 bit 7.10 on my desktop after running the 32 bit version for the last 5 months. the only thing that gave me trouble was configuring xorg.conf to use both my monitors. I finally got it working after about 4 hours of googling.


here is my xorg.conf file.

```
# Fallback X.org config file
# FIXME: include the wacom devices (would only add noise at the current
#        development stage)

Section "InputDevice"
        Identifier      "Generic Keyboard"
        Driver          "kbd"
        Option          "CoreKeyboard"
        Option          "XkbRules"      "xorg"
        Option          "XkbModel"      "pc105"
        Option          "XkbLayout"     "ch"
        Option          "XkbVariant"    "de_nodeadkeys"
EndSection

Section "InputDevice"
	Identifier	"Configured Mouse"
	Driver		"mouse"
	Option		"CorePointer"
	Option		"Device"	"/dev/input/mice"
	Option		"Protocol"	"ImPS/2"
	Option		"ZAxisMapping"	"4 5"
	Option		"Emulate3Buttons"	"false"
EndSection

Section "ServerLayout"
	Identifier	"Default Layout"
	Inputdevice	"Generic Keyboard"
	Inputdevice	"Configured Mouse"
  screen 0 "screen1" 0 0
EndSection

Section "Module"
	Load		"glx"
	Load		"v4l"
EndSection

Section "device"
	Identifier	"device1"
	Boardname	"NVIDIA GeForce 8 Series"
	Busid		"PCI:6:0:0"
	Driver		"nvidia"
	Screen	0
	Vendorname	"NVIDIA"
EndSection

Section "screen"
	Identifier	"screen1"
	Device		"device1"
	Defaultdepth	24
	Monitor		"monitor1"
	Option		"RenderAccel"	"True"
	Option		"TwinView"	"1"
	Option		"MetaModes"	"1280x1024,1280x1024"
	Option		"TwinViewOrientation"	"RightOf"
	SubSection "Display"
		Depth	24
		Modes		"1280x1024@60"	"1024x768@60"	"800x600@60"	"640x480@60"
	EndSubSection
EndSection

Section "monitor"
	Identifier	"monitor1"
	Vendorname	"Dell"
	Modelname	"Dell E196FP"
	Horizsync	31.0-83.0
	Vertrefresh	56.0-76.0
  modeline  "640x480@60" 25.2 640 656 752 800 480 490 492 525 -vsync -hsync
  modeline  "800x600@60" 40.0 800 840 968 1056 600 601 605 628 +hsync +vsync
  modeline  "1024x768@60" 65.0 1024 1048 1184 1344 768 771 777 806 -vsync -hsync
  modeline  "1280x1024@60" 108.0 1280 1328 1440 1688 1024 1025 1028 1066 +hsync +vsync
	Gamma	1.0
EndSection

Section "ServerFlags"
	Option		"Xinerama"	"0"
EndSection
```

The line that finally did it for me was the MetaModes:
```
	Option		"MetaModes"	"1280x1024,1280x1024"
```
