### Post by Dellfan on 2007-10-20
I would suggest:

Section "Device"
        Identifier      "nVidia Corporation G72M [Quadro NVS 110M/GeForce Go 7300]"
        Boardname       "nv"
        Busid           "PCI:1:0:0"
        Driver          "nvidia"
        Screen  0
        Option          "AddARGBVisuals"        "True"
        Option          "AddARGBGLXVisuals"     "True"
        Option          "NoLogo"                "True"
        Option          "NvAGP"
EndSection

Section "Monitor"
        Identifier      "Dell LCD 1440x900"
        Modelname       "Custom 1"
        HorizSync       30-83
        #VertRefresh    50-76
        VertRefresh     60
        DisplaySize     333 212
        #1440x900 @ 60hz
        modeline    "1440x900" 108.84 1440 1472 1880 1912 900 918 927 946 +hsync +vsync
        #1280x1024 @ 60hz
        modeline "1280x1024" 108.000 1280 1328 1440 1688 1024 1025 1028 1066 +hsync +vsync
        Gamma   1.0
EndSection

Section "Screen"
        Identifier      "Default Screen"
        Device          "nVidia Corporation G72M [Quadro NVS 110M/GeForce Go 7300]"
        Monitor         "Dell LCD 1440x900"
        Defaultdepth    24
        SubSection "Display"
                Depth   24
                Virtual 1440    900
                Modes "1440x900" "1280x1024" "1280x800" "1152x768" "1024x768"
        EndSubSection
EndSection
