### Post by KdotJ on 2010-08-19
Have you set a specific layout manager to the JFrame or JTabbedPane? If not, it will be default to BorderLayout (or is it FlowLayout?) either way, the JPanel itself will not do anything if you resize the frame. The GridBagLayout only applies to components inside the JPanel in your case. The behaviour of the JPanel, regarding the panel itself, is determined by the layout manager of the container the panel is placed in.

[PHP]
public class Arrow extends JPanel{
	public enum Vector {UP, RIGHT, INTER};
	private int arrowLength=10;
	private Color background=Color.BLACK, lines=Color.WHITE;
	private Vector vec;
	
	
	public Arrow(Vector vec){
		this.vec=vec;
		this.setBackground(background);
	}
	
	public Arrow(Vector vec, Color lines, Color background){
		this(vec);
		this.lines=lines;
		this.background=background;
	}
	
	public Arrow(Vector vec, Color lines, Color background, int arrowLength){
		this(vec, lines, background);
		this.arrowLength=arrowLength;
	}
	
	public void paintComponent(Graphics g){
		super.paintComponent(g);
		g.setColor(lines);
		switch(vec){
		case UP:	arrowUp(g);
					break;
		case RIGHT:	arrowRight(g);
					break;
		case INTER:	intersect(g);
					break;
		}
	}
	
	//Right
	private void arrowRight(Graphics g){
		int length=this.getWidth(), height = this.getHeight(), height_=height/2, aLength=length-arrowLength;
		g.drawLine(0, height_, length, height_);
		g.drawLine(aLength, 0, length, height_);
		g.drawLine(aLength, height, length, height_);
	}
	
	//Up
	private void arrowUp(Graphics g){
		int length=this.getHeight(), width=this.getWidth(), width_=width/2;
		g.drawLine(width_, 0, width_, length);
		g.drawLine(0, arrowLength, width_, 0);
		g.drawLine(width, arrowLength, width_, 0);
	}
	
	private void intersect(Graphics g){
		int height = this.getHeight(), height_ = height/2, width=this.getWidth(), width_=width/2;
		g.drawLine(0,height_,width, height_);
		g.drawLine(width_, 0, width_, height);
	}
}

### Post by kahumba on 2010-08-20
It looks like you're trying to create a vector-based diagram, if so, creating a new JPanel for x & y arrows isn't a flexible/good idea imho, especially as it comes to actually drawing your data.
Here's my solution I quickly created for you. Screenshots "old.png"  is your app, "new*" are mine - at normal at resized size, notice that the diagram updates accordingly, it also treats the coordinates as expected: x, y start from the lower left corner and since it's "vector based" the coordinates system can go from (x,y) 0, 0 to 100, 100.
Here's the code

Main.java (the window):
```

package test;

import java.awt.*;
import java.util.ArrayList;
import javax.swing.*;

public class Main extends JFrame {
    public static void main(String[] args) {
        new Main();
    }

    public Main() {
        setDefaultCloseOperation(EXIT_ON_CLOSE);
        setSize(500, 500);
        createComponents();

        Dimension dim = Toolkit.getDefaultToolkit().getScreenSize();
        setLocation(dim.width/2-getWidth()/2, dim.height/2 - getHeight()/2);
        setVisible(true);
    }

    private void createComponents() {
        JTabbedPane tabs = new JTabbedPane();
        setContentPane(tabs);
        Color[] colors = {Color.black, Color.black, Color.red};
        Diagram diagram = new Diagram(Color.white, colors);
        tabs.addTab("Diagram #1", diagram);

        //add some data
        ArrayList<Point> list = new ArrayList<Point>();
        list.add(new Point(0, 0));
        list.add(new Point(20, 20));
        list.add(new Point(40, 20));
        list.add(new Point(50, 60));
        list.add(new Point(70, 50));
        list.add(new Point(80, 60));
        diagram.setData(list);
    }
}

```Diagram.java (the class which renders the diagram):
```

package test;

import java.awt.*;
import java.util.ArrayList;
import javax.swing.*;

public class Diagram extends JPanel {
    private Color[] colors;
    private ArrayList<Point> list = new ArrayList<Point>();
    private static final int off = 10;
    private static final BasicStroke thick = new BasicStroke(2.5f);
    private static final BasicStroke thin = new BasicStroke(1.0f);

    public Diagram(Color background, Color[] color) {
        this.setBackground(background);
        this.colors = color;
    }

    public void setData(ArrayList<Point> list) {
        this.list = list;
        repaint();
    }

    @Override
    public void paint(Graphics gr) {
        if(gr == null) {
            return;
        }

        Graphics2D g = (Graphics2D)gr;
        g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
        g.setRenderingHint(RenderingHints.KEY_RENDERING, RenderingHints.VALUE_RENDER_QUALITY);

        int w = getWidth(), h = getHeight();

        g.setColor(getBackground());
        g.fillRect(0, 0, w, h);

        //left vert arrow
        g.setStroke(thick);
        g.setColor(colors[0]);
        
        final int arrowSize = off/2;
        g.drawLine(off, off, off, h-off);
        g.drawLine(off, off, arrowSize, off*2);
        g.drawLine(off, off, off+arrowSize, off*2);

        //right bottom horiz arrow
        g.setColor(colors[1]);
        final int y = h-off;
        g.drawLine(off, y, w-off, y);
        g.drawLine(w-off, y, w-off-arrowSize, y-arrowSize);
        g.drawLine(w-off, y, w-off-arrowSize, y+arrowSize);


        //draw data
        if(list.size() < 2) {
            return;
        }
        g.setStroke(thin);
        g.setColor(colors[2]);
        Point _p1 = normalize(list.get(0));
        Point _p2 = normalize(list.get(1));
        g.drawLine(_p1.x, _p1.y, _p2.x, _p2.y);

        for(int i=1; i<list.size()-1; i++) {
            Point p1 = normalize(list.get(i));
            Point p2 = normalize(list.get(i+1));
            g.drawLine(p1.x, p1.y, p2.x, p2.y);
        }
    }

    private Point normalize(Point p) {
        float x = p.x, y = p.y;
        int w = getWidth(), h = getHeight();
        x = (w*x)/100;
        y = (h*y)/100;
        x = off + x;
        y = h - y - off;
        return new Point((int)x, (int)y);
    }
}
