### Post by diesch on 2009-11-14
Using regular expressions to parse HTML only works for simple cases.

Maybe a simple
```
  lynx -dump -listonly -force_html SOME_FILE_OR_URL
```
does the job. Otherwise use a HTML parse. For python there's [BeautifulSoup](http://www.crummy.com/software/BeautifulSoup/) (package python-beautifulsoup).

### Post by doas777 on 2009-11-14
ok, heres what I do, but may not be the "best way"

1) download teh html
2) open it, and get the file lines
3) search for an attribute that defines teh line I;m looking for 
4) search the line I found for the start delimiter of the value i want,
5)snag the value from the start delimiter to the end delimiter.
6) rinse and repeat as needed.

here is an excerpt from my python manga grabber:
```

class SearchCriteria:
    def __init__(self, iLineContains, iTokenStartsAfter, iSplitStr):
        self.lineContains = iLineContains
        self.tokenStartsAfter = iTokenStartsAfter
        self.splitStr = iSplitStr

class PageParser:
    def __init__(self):
        self._filepath = ""            

    def download_page(self, iUrl, iDest):
        try:
            if not os.path.exists(os.path.dirname(iDest)):
                os.makedirs(os.path.dirname(iDest), 0777)
            if os.path.exists(iDest):
                os.remove(iDest)
                #print "overwriting file %s" %(iDest)
    
            urllib.urlretrieve(iUrl.replace("\"", ""), iDest)
            self._filepath = iDest
        except Exception, e:
            print "Exception: ", e

    def read_file(self):
        ls = {}
        with open(self._filepath) as f:
            ls = f.readlines()
        return ls
            
    def parse_for_value(self, iSearchCriteria):
        # @type iSearchCriteria SearchCriteria
        t = {}
        value = "" #contains return if found, "" if not.
        ls = self.read_file()
        for l in ls:
            #print l
            if l.find(iSearchCriteria.lineContains) >= 0:
                t = l.split(iSearchCriteria.splitStr)
                break

        if len(t) > 0:
            for s in t:
                if s.startswith(iSearchCriteria.tokenStartsAfter):
                    value = s.lstrip(iSearchCriteria.tokenStartsAfter)
                    break
        else: 
            print "no line found containing %s" %(iSearchCriteria.lineContains)
        return value
