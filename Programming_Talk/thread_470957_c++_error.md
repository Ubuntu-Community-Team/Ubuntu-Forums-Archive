```

class window : Gtk::Window
{

public:
     window();
     ~window();

     void on_button1_clicked();

window.cpp
```

     window::window()
     {
          // Create window
          
          // Attach
         button1.signal_clicked().connect( sigc::ptr_fun( &on_button1_clicked() ) );
     }
     window::~window()
     {
     }

     window::void on_button1_clicked()
     {
         cout << "Button1 was clicked! Yay!\n";
     }
