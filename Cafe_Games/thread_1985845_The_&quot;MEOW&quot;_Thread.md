 {

std::bool isDog(Meow meowthread)
{
// check if there are dogs in the thread
  int post = 1;
  do
 {
    if(meowthread.postnum(post) == meowthread.dog)
    {
      return(true);
    }
   ++post;
  }
while(post < meowthread.get_num_posts());

}
