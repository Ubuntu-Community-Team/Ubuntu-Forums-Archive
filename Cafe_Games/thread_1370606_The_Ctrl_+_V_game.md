Public Function Encrypt(Password As String) As String
    Dim adStm As ADODB.Stream

    Dim strFib As String

    Dim strFinChars As String
    'Setup the stream
    Set adStm = New ADODB.Stream
    adStm.Mode = adModeReadWrite
    adStm.Type = adTypeText
    adStm.Charset = "ascii"
    adStm.Open
    adStm.WriteText Password
    
    Dim f1, f2, f As Long
    f2 = 5
    f1 = 8
    
    Encrypt = ""
    
    adStm.Position = 0
    
    Do While Not adStm.EOS
        f = f1 + f2
        f2 = f1
        f1 = f
        
        strFib = strFib & f & ", "
        intCharNum = Asc(adStm.ReadText(1))
        strReadChars = strReadChars & intCharNum & ", "
        intCharNum = intCharNum + f
