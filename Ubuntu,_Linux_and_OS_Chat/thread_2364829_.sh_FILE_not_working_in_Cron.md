### Post by yp.2297 on 2017-06-28
I got following script from net. I changes somethings. and now it's not working on crontab. it's working good on terminal.

Crontab:: 
30 2 * * * sh /var/www/html/final.sh

following is the script::
#!/bin/bash
main(){
init
chkBackupDir
}
 
init() {
BACKUP_DIR="/var/www/html/backups"
DB_USER="wordpressU"
DB_PASSWORD="wordpressP"
DB_NAME="wordpress"
DOC_ROOT="/var/www/html/wordpress"
DAYS_TO_KEEP_BACKUP="5"
}
 
chkBackupDir() {
if [ ! -d "$BACKUP_DIR" ]; then
mkdir "$BACKUP_DIR"
fi
bakupDocRoot
}
 
bakupDocRoot() {
`which tar` -cjf "$BACKUP_DIR"/doc_root_backup-$(date +%d-%m-%y).tar.bz2 --directory="$DOC_ROOT" .
if [[ -e "$BACKUP_DIR"/doc_root_backup-$(date +%d-%m-%y).tar.bz2 ]]; then
bakupDatabase
else
echo "Backup failed for document root on your server. Please check manually"
fi
}


bakupDatabase() {
`which mysqldump` -u "$DB_USER" -p"$DB_PASSWORD" "$DB_NAME" > "$BACKUP_DIR"/"$DB_NAME"-$(date +%d-%m-%y).sql | tee "$BACKUP_DIR"/backup.log
tar -cjf "$BACKUP_DIR"/"$DB_NAME"-$(date +%d-%m-%y).tar.bz2 "$BACKUP_DIR"/"$DB_NAME"-$(date +%d-%m-%y).sql --remove-files 2> $BACKUP_DIR/tar.log
if [[ -e "$BACKUP_DIR"/"$DB_NAME"-$(date +%d-%m-%y).tar.bz2 ]]; then
createTar
else
echo "Backup failed for database on your server. Please check manually" 
fi
}
 
createTar() {
cd "$BACKUP_DIR"
`which tar` -cf backup-$(date +%d-%m-%y).tar doc_root_backup-$(date +%d-%m-%y).tar.bz2 "$DB_NAME"-$(date +%d-%m-%y).tar.bz2 --remove-files 2> final.log
deleteOldBackup
}
 deleteOldBackup() {
`which find` "$BACKUP_DIR"/ -mtime +"$DAYS_TO_KEEP_BACKUP" -exec rm {} \; >> "$BACKUP_DIR"/delete.log
}


main
# End



what is the problem??

### Post by spjackson on 2017-06-28
Welcome to the forums.

Your script declares itself to be a bash script
> 
#!/bin/bash

which is good, because it uses bash syntax. But your crontab entry specifically requests 'sh' to interpret the script.
> 
sh /var/www/html/final.sh

and that's bad because the script is a bash script. One way to correct this is to change your crontab entry to specify bash.
```

30 2 * * * bash /var/www/html/final.sh

```
