### Post by blastus on 2007-04-29
I had to add the NoDFPNativeResolutionCheck to get X to use my custom ModeLine in Feisty. I've used the same xorg.conf configuration since Breezy with no problems but in Feisty I had to add the NoDFPNativeResolutionCheck. Now my xorg.conf is getting increasingly complicated but it works :) 

```
Section "Monitor"
    Identifier     "Generic Monitor"
    HorizSync       30.0 - 81.0
    VertRefresh     56.0 - 75.0
    Option         "DPMS"
    # 1280x1024, 75.0Hz; hfreq=79.98, vfreq=75.03
    ModeLine "1280x1024_x" 135.00 1280 1296 1440 1688 1024 1025 1028 1066 +hsync +vsync
EndSection

Section "Device"
    Identifier     "NVIDIA Corporation NV31 [GeForce FX 5600]"
    Driver         "nvidia"
    BusID	   "PCI:1:0:0"
    Option         "AddARGBGLXVisuals" "True"
    Option         "UseEDID" "False"
    # To fix tearing artifacts in xine full-screen mode we use a TwinView hack
    # In xine, make sure the video driver is set to xv
    # NOTE: TwinView will cause the X to report the refresh rate as 50hz but
    #       don't worry about it--it is actually 75hz
    #       TwinView will also enable you to take screenshots with overlay video!
    # Works with NVIDIA Driver Version 1.0-8762+
    Option         "TwinView"
    Option         "TwinViewOrientation" "Clone"
    Option         "MetaModes" "DFP-0: 1280x1024_x, CRT-0: 1280x1024_x"
    Option         "ConnectedMonitor" "DFP-0, CRT-0"
EndSection

Section "Screen"
    Identifier     "Default Screen"
    Device         "NVIDIA Corporation NV31 [GeForce FX 5600]"
    Monitor        "Generic Monitor"
    DefaultDepth    24
    Option         "ExactModeTimingsDVI" "TRUE" 
    Option         "ModeValidation" "DFP-0: AllowNon60HzDFPModes, NoEdidModes, NoVesaModes, NoXServerModes, NoDFPNativeResolutionCheck"    
    SubSection     "Display"
        Depth       24
        Modes       "1280x1024_x"
    EndSubSection
EndSection
```
