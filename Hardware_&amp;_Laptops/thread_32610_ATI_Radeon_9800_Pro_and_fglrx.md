### Post by StacyWebb on 2005-05-10
the only way I know of is to actually edit the xorg.conf file directly by adding what resolution you wish for.
 I currently run at 1600x1050 on ati.

Partial of my xorg.conf

> Section "Screen"
        Identifier      "Default Screen"
        Device          "ATI Technologies, Inc. RV350 [Mobility Radeon 9600 M10]"
        Monitor         "Generic Monitor"
        DefaultDepth    24
        SubSection "Display"
                Depth           1
                Modes           "1680x1050"
        EndSubSection

EndSection

Also a better test for fglrx is fgl_glxgears this will give you a better idea of the fps.
If you are running 8.12 for fglrx this should be up in the 400s

### Post by cain171562 on 2008-01-26
I have a Radeon 9800 Pro 128MB and I finally got compiz running with a good resolution 1280x1024.  My GlxGears is around 6500 This is with the ATI Binary Driver.  Below is my xorg.conf if it helps anyone Its a little messy and theres probably some stuff in there I dont need, but its working.


# xorg.conf (xorg X Window System server configuration file)
#
# This file was generated by failsafeDexconf, using
# values from the debconf database and some overrides to use vesa mode.
#
# You should use dexconf or another such tool for creating a "real" xorg.conf
# For example:
#   sudo dpkg-reconfigure -phigh xserver-xorg

Section "ServerLayout"
	Identifier     "Default Layout"
	Screen      0  "aticonfig-Screen[0]" 0 0
	InputDevice    "Generic Keyboard"
	InputDevice    "Configured Mouse"
EndSection

Section "Files"
EndSection

Section "Module"
	Load  "dbe"
	Load  "dri"
	Load  "glx"
	Load  "vbe"
	Load  "GLcore"
	Load  "extmod"
    	Load  "ddc
    	Load  "type1"	
   	Load  "bitmap"
EndSection


Section "InputDevice"
	Identifier  "Generic Keyboard"
	Driver      "kbd"
	Option	    "CoreKeyboard"
	Option	    "XkbRules" "xorg"
	Option	    "XkbModel" "pc105"
	Option	    "XkbLayout" "us"
EndSection

Section "InputDevice"
	Identifier  "Configured Mouse"
	Driver      "mouse"
	Option	    "CorePointer"
	Option	    "Device" "/dev/input/mice"
	Option	    "Protocol" "ExplorerPS/2"
	Option	    "ZAxisMapping" "6 7"
	Option	    "Emulate3Buttons" "true"
EndSection

Section "Monitor"
	Identifier   "Failsafe Monitor"
	Option	    "DPMS"
EndSection

Section "Monitor"
	Identifier   "aticonfig-Monitor[0]"
	Option	    "VendorName" "ATI Proprietary Driver"
	Option	    "ModelName" "Generic Autodetecting Monitor"
	Option	    "DPMS" "true"
EndSection

Section "Device"
	Identifier  "Failsafe Device"
	Driver      "fglrx"
	BusID       "PCI:1:0:0"
EndSection

Section "Device"
	Identifier  "aticonfig-Device[0]"
	Driver      "fglrx"
	Option "HWCursor" "On"
    	Option "CapabilitiesEx" "0×00000000"
    	Option "XAANoOffscreenPixmaps" "true"
    	Option "mtrr" "off"
    	Option "no_accel" "no"
    	Option "BlockSignalsOnLock" "on"
    	Option "KernelModuleParm" "locked-userpages=0"
    	Option "UseFastTLS" "0"
    	Option "RenderAccel" "true"
    	Option "AllowGLXWithComposite" "true"



EndSection

	Identifier "Default Screen"
	Device     "Failsafe Device"
	Monitor    "Failsafe Monitor"
	DefaultDepth     24
	SubSection "Display"
		Depth     24
		Modes    "1280x1024" "800x600"
	EndSubSection
EndSection
