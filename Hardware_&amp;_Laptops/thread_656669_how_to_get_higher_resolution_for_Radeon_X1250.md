### Post by intel on 2008-01-02
This is a known issue with fglrx 7.12,
at this point your only options are to

[LIST]
[*]wait for new fglrx version
[*]downgrade to fglrx 7.11[/LIST]or wait for new fglrx version while using radeonHD,

for all X1250 related problems are discussed here with solutions:-
[https://groups.google.com/group/x1250](https://groups.google.com/group/x1250)

I don't know if it'll help any, but here's the xorg.conf I use for fglrx.
```
Section "Modules"
	Load	"dri"
	Load	"glx"
EndSection

Section "Files"
EndSection

Section "InputDevice"
	Identifier	"Generic Keyboard"
	Driver		"kbd"
	Option		"CoreKeyboard"
	Option		"XkbRules"	"xorg"
	Option		"XkbModel"	"pc105"
	Option		"XkbLayout"	"us"
EndSection

Section "InputDevice"
	Identifier	"Configured Mouse"
	Driver		"mouse"
	Option		"CorePointer"
	Option		"Device"		"/dev/input/mice"
	Option		"Protocol"		"ImPS/2"
	Option		"ZAxisMapping"		"4 5"
	Option		"Emulate3Buttons"	"true"
EndSection

Section "Device"
	Identifier	"Generic Video Card"
	Driver		"fglrx"
	BusID		"PCI:1:5:0"
	Option	   	"VideoOverlay" "on"
	Option	    	"OpenGLOverlay" "off"
EndSection

Section "Monitor"
	Identifier	"VX2025wm"
	Option		"DPMS"
EndSection

Section "Screen"
	Identifier	"Default Screen"
	Device		"Generic Video Card"
	Monitor		"VX2025wm"
	DefaultDepth	24
	SubSection "Display"
		Modes		"1680x1050" "1280x1024" "1280x960" "1024x768" "800x600" "640x480"
	EndSubSection
EndSection

Section "ServerLayout"
	Identifier	"Default Layout"
	Screen		"Default Screen"
	InputDevice	"Generic Keyboard"
	InputDevice	"Configured Mouse"
EndSection

Section "DRI"
	Mode 0666
EndSection

Section "Extensions"
	Option "Composite" "Off"
EndSection

Section "ServerFlags"
	Option "AIGLX" "Off"
EndSection
```
