### Post by artio on 2005-03-29
Eric,

I've had the same problem, first in Red Hat 9 and now in Hoary. The only way I've gotten my external 1280x1024 flat panel to work with the native 1024x768 laptop is to adjust the xorg.conf so as to include two identical "Monitor" and "Screen" entries refelcting the specs for the external monitor, and use Option "MonitorLayout" "LFP,CRT" in the Device and Monitor sections (see below). 

Thisa approach seems to make no sense but any more sensical options have not worked. Input from more knowledgeable people would be greatly appreciated.

...

Section "Device"
	Identifier	"Intel 82852/855GM"
	Driver		"i810"
	BusID		"PCI:0:2:0"
	VideoRam	32768
	Option "MonitorLayout" "LFP,CRT"
	Option "DisplayInfo" "FALSE"
EndSection

Section "Monitor"
	Identifier "Monitor0"
	VendorName "Monitor Vendor"
	ModelName "L1730S"
	HorizSync 30.0 - 83.0
	VertRefresh 56.0 - 75.0
	Option "dpms"
EndSection

Section "Monitor"
	Identifier "Monitor1"
	VendorName "LG"
	ModelName "L1730S"
	HorizSync 30.0 - 83.0
	VertRefresh 56.0 - 75.0
	Option "dpms"
EndSection

Section "Screen"
	Identifier "Screen0"
	Device "Intel 82852/855GM"
	Monitor "Monitor0"
	DefaultDepth 16
	SubSection "Display"
		Depth 16
		Modes "1280x1024"
	EndSubSection
EndSection

Section "Screen"
	Identifier "Screen1"
	Device "Intel 82852/855GM"
	Monitor "Monitor1"
	DefaultDepth 16
	SubSection "Display"
		Depth 16
		Modes "1280x1024"
	EndSubSection
EndSection

Section "ServerLayout"
	Identifier	"Default Layout"
	Screen	0	"Screen0" 0 0
	InputDevice	"Generic Keyboard"
	InputDevice	"Configured Mouse"
	InputDevice	"Synaptics Touchpad"
	Option 		"MonitorLayout" "LFP,CRT"
EndSection

Section "DRI"
	Mode	0666
EndSection
