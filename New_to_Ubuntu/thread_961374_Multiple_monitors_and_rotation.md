### Post by aeiah on 2008-10-28
have you tried using nvidia-settings? its rather intuitive and should give you the options you need, especially with regards to rotation.

as for the primary issue, if nvidia-settings doesnt solve it, then i guess you'll have to fiddle with xorg.conf. as i understand it, your right monitor is using your 5600 card and the other two are using your 8800? make sure the screen, display and videocard numbers all add up i guess. they look like they do in xorg, although you havent specified screen 2 in your ```
Section "Device"
Identifier "Videocard2"
```
section like you have in the other paragraphs.

you could also look into whether your BIOS is set to use the 5600 as primary. perhaps xorg is reading the information from there.

### Post by CompShrink on 2008-11-28
I had the same issue when I installed 8.10, the "Rotation Settings" menu on the left in Nvidia settings didn't appear. In 8.04 it was there. Even after copying the old xorg.conf file, it wasn't there. I guess the new driver sometimes mistakenly detects your card doesn't support rotation? Both 173 and 177 don't show this option for me.

The good news is, I finally found an option that worked for me. I enabled xinerama (and NOT twinview) in the nvidia-settings and added to the Screen sections of only the monitors I wanted rotated the following line:
    Option         "Rotate" "left"

Here's my xorg.conf, hopefully a similar setup will work for you.

Section "ServerLayout"
    Identifier     "Default Layout"
    Screen      0  "Screen0" 0 0
    Screen      1  "Screen1" RightOf "Screen0"
    InputDevice    "Generic Keyboard" "CoreKeyboard"
    InputDevice    "Configured Mouse" "CorePointer"
EndSection

Section "Files"
EndSection

Section "Module"
    Load           "glx"
    Load           "v4l"
EndSection

Section "ServerFlags"

# Removed Option "Xinerama" "0"
    Option         "Xinerama" "1"
EndSection

Section "Monitor"
    Identifier     "monitor1"
    VendorName     "Unknown"
    ModelName      "Acer X193W+"
    HorizSync       30.0 - 82.0
    VertRefresh     56.0 - 76.0
    Gamma           1
EndSection

Section "Monitor"
    Identifier     "Monitor0"
    VendorName     "Unknown"
    ModelName      "Acer X193W+"
    HorizSync       30.0 - 82.0
    VertRefresh     56.0 - 76.0
EndSection

Section "Device"
    Identifier     "device1"
    Driver         "nvidia"
    VendorName     "NVIDIA Corporation"
    BoardName      "GeForce 7300 GT"
    BusID          "PCI:1:0:0"
    Screen          1
EndSection

Section "Device"
    Identifier     "Videocard0"
    Driver         "nvidia"
    VendorName     "NVIDIA Corporation"
    BoardName      "GeForce 7300 GT"
EndSection

Section "Device"
    Identifier     "Device0"
    Driver         "nvidia"
    VendorName     "NVIDIA Corporation"
    BoardName      "GeForce 7300 GT"
    BusID          "PCI:1:0:0"
    Screen          0
EndSection

Section "Screen"
    Identifier     "screen1"
    Device         "Device1"
    Monitor        "Monitor1"
    DefaultDepth    24
#    Option         "Rotate" "left"
    Option         "TwinView" "0"
    Option         "metamodes" "CRT: nvidia-auto-select +0+0"
    SubSection     "Display"
        Depth       24
    EndSubSection
EndSection

Section "Screen"
    Identifier     "Screen0"
    Device         "Device0"
    Monitor        "Monitor0"
    DefaultDepth    24
    Option         "Rotate" "left"
    Option         "TwinView" "0"
    Option         "metamodes" "DFP: nvidia-auto-select +0+0"
    SubSection     "Display"
        Depth       24
    EndSubSection
EndSection
