### Post by ViperKnight on 2007-05-22
Here's my .asoundrc if you want it.  I have the same onboard chip and using a 5.1 speaker setup.
This file does channel mixing (spreads stereo sounds too all speakers) as well as Dmixing (lets you use multiple ALSA programs at the same time).

Just make sure you use "default" as your sound device in your programs and you're set;)

You can find the file using Nautilus by pressing "Ctrl+H" to unhide files and look for it in your home directory.  If it doesnt exist just create a new document and paste what's below into there and save it as ".asoundrc" (no extension).

```

###########33
pcm.snd_card {
     type hw
     card 0 # change to your cards number or name
}

# 6 channel dmix:
pcm.dmix6 {
     type dmix
        ipc_key 1024
        ipc_key_add_uid false # let multiple users share
        ipc_perm 0660 # IPC permissions (octal, default 0600)
        slave {
                pcm snd_card # see below
                rate 44100
                channels 6
		period_time 0
		period_size 1024
		buffer_time 0
		buffer_size 4096               # in case of skipping, try increasing
        }
     }

# upmixing: 
pcm.ch51dup {
        type route
        slave.pcm dmix6
        slave.channels 6
        ttable.0.0 1
        ttable.1.1 1
        ttable.0.2 1
        ttable.1.3 1
        ttable.0.4 0.5
        ttable.1.4 0.5
        ttable.0.5 0.5
        ttable.1.5 0.5
   }

pcm.duplex {
     type asym
     playback.pcm "ch51dup" # upmix first
#     playback.pcm "dmix6"  # just pass to 6 channel dmix
#     capture.pcm "dsnoop:0" # doesn't work for me
     capture.pcm "snd_card"
}

# change default device:
pcm.!default {
     type plug
     slave.pcm "duplex"
}

# for aoss
pcm.dsp "duplex"

pcm.dsp1 "duplex"

```

EDIT: I also fixed issues with choppy sound when using ALSA in that file^^ which seemed to be common with the ALC850 chips because of the available buffer size.
