### Post by ViperKnight on 2007-04-16
Find the Alsa wiki and search for Surround.
You'll have to modify your .asoundrc file in order to get your soundcard to duplicate the channels for full 4.1 sound from stereo files.  I think the Alsa wiki actually have a preset file for the Intel Integrated sound cards.

Here's my .asoundrc but it's designed for 5.1 on my Realtek ALC850(comes up as Nvidia CK804) onboard audio.  The "upmixing" section is where the soundcard is told to duplicate channels across 5.1 ;)

Make sure you backup your old one before using a new one or modifying ;)
```

###########33
pcm.snd_card {
     type hw
     card 0 # change to your cards number or name
}

# 6 channel dmix:
pcm.dmix6 {
     type dmix
        ipc_key 1024
        ipc_key_add_uid false # let multiple users share
        ipc_perm 0660 # IPC permissions (octal, default 0600)
        slave {
                pcm snd_card # see below
                rate 44100
                channels 6
		period_time 0
		period_size 1024
		buffer_time 0
		buffer_size 4096               # in case of skipping, try increasing
        }
     }

# upmixing: 
pcm.ch51dup {
        type route
        slave.pcm dmix6
        slave.channels 6
        ttable.0.0 1
        ttable.1.1 1
        ttable.0.2 1
        ttable.1.3 1
        ttable.0.4 0.5
        ttable.1.4 0.5
        ttable.0.5 0.5
        ttable.1.5 0.5
   }

pcm.duplex {
     type asym
     playback.pcm "ch51dup" # upmix first
#     playback.pcm "dmix6"  # just pass to 6 channel dmix
#     capture.pcm "dsnoop:0" # doesn't work for me
     capture.pcm "snd_card"
}

# change default device:
pcm.!default {
     type plug
     slave.pcm "duplex"
}

# for aoss
pcm.dsp "duplex"

pcm.dsp1 "duplex"

```
