### Post by Roderik on 2008-01-22
I'm trying to package a very simple mono program. It's just compiling 1 .exe and putting it in /usr/bin. Not too hard, or so i thought...

So my rules file just compiles the exe (one liner in build) and should install it in /usr/bin. 

```

#!/usr/bin/make -f

CC = gcc
CFLAGS = -g -Wall
INSTALL_PROGRAM = install

ifeq (,$(findstring noopt,$(DEB_BUILD_OPTIONS)))
  CFLAGS += -O2
endif
ifeq (,$(findstring nostrip,$(DEB_BUILD_OPTIONS)))
  INSTALL_PROGRAM += -s
endif

build:
        $(checkdir)
        mkdir -p $$(pwd)/debian/tmp/usr/bin/
        gmcs -out:$$(pwd)/debian/tmp/usr/bin/mmcl.exe -unsafe Main.cs General.cs AVIFile.cs AVIModifier.cs MPEG4FrameModifier.cs
        touch build

clean:
        $(checkdir)
        rm -f build
        rm -rf *~ debian/tmp debian/*~ debian/files* debian/substvars

binary-indep:   checkroot build
        $(checkdir)

# There are no architecture-independent files to be uploaded
# generated by this package.  If there were any they would be
# made here.

binary-arch:    checkroot build
        $(checkdir)
        install -m 755 -T debian/tmp/usr/bin/mmcl.exe  /usr/bin/mmcl.exe
        install -d debian/tmp/DEBIAN $(docdir)
        dpkg-gencontrol -isp
        chown -R root:root debian/tmp
        chmod -R u+w,go=rX debian/tmp
        dpkg --build debian/tmp ..

define checkdir
endef

binary: binary-indep binary-arch

checkroot:
        $(checkdir)
        test $$(id -u) = 0

.PHONY: binary binary-arch binary-indep clean checkroot

```
