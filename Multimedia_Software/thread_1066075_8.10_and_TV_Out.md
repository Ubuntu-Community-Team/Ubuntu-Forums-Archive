### Post by NT4usB on 2009-02-10
Run: ```
sudo nvidia-settings
```
There, you should be able to add a xscreen and configure it seperately for the TV. 
Should add more stuff to the xorg.conf. 
You can manually add/edit using the Save xorg.conf>preview changes option.

Most of this should already be in there... Modes, etc.
```
Section "Screen"
Identifier "whatever your _TV_ screen is"
stuff...
stuff...
TVStandard "HD480p"
TVOutFormat "Component" #you are using RGB connection, aren't you?
SubSection
stuff...
stuff...
EndSubSection
EndSection
```

### Post by NT4usB on 2009-02-11
Don't know what HDCP is...
I have a 8-9 year old Panasonic tube connected by component that works great. Only supports HD480i via component. 
Here's a xorg.conf that worked on it:```
Section	"ServerLayout"
    Identifier     "Simple Layout"
    Screen      0  "Screen[0]" 0 0
    InputDevice    "Generic Keyboard" "CoreKeyboard"
    InputDevice    "Configured Mouse" "CorePointer"
EndSection

Section "Files"
    FontPath        "/usr/share/X11/fonts/misc"
    FontPath        "/usr/share/X11/fonts/cyrillic"
    FontPath        "/usr/share/X11/fonts/100dpi/:unscaled"
    FontPath        "/usr/share/X11/fonts/75dpi/:unscaled"
    FontPath        "/usr/share/X11/fonts/Type1"
    FontPath        "/usr/share/X11/fonts/100dpi"
    FontPath        "/usr/share/X11/fonts/75dpi"
    FontPath        "/var/lib/defoma/x-ttcidfont-conf.d/dirs/TrueType"
EndSection

Section "Module"
    Load           "i2c"
    Load           "bitmap"
    Load           "ddc"
    Load           "extmod"
    Load           "freetype"
    Load           "glx"
    Load           "int10"
    Load           "type1"
    Load           "vbe"
EndSection

Section "InputDevice"
    Identifier     "Generic Keyboard"
    Driver         "kbd"
    Option         "CoreKeyboard"
    Option         "XkbRules" "xorg"
    Option         "XkbModel" "pc104"
    Option         "XkbLayout" "us"
EndSection

Section "InputDevice"
        Identifier      "Configured Mouse"
        Driver          "mouse"
        Option          "CorePointer"
        Option          "Device"                "/dev/input/mice"
        Option          "Protocol"              "ExplorerPS/2"
        Option          "Buttons"               "8"
        Option          "ZAxisMapping"          "4 5"
        Option          "Emulate3Buttons"       "false"
        Option          "ButtonMapping"         "1 6 3 2 4 5"
        Option          "Resolution"            "100"
EndSection

Section "Monitor"
    Identifier     "Monitor[0]"
    HorizSync       30.0 - 50.0
    VertRefresh     60.0 - 60.0
EndSection

Section "Device"
    Identifier     "Device[0]"
    Driver         "nvidia"
    BusID          "PCI:1:0:0"
    Screen          0
EndSection

Section "Screen"
    Identifier     "Screen[0]"
    Device         "Device[0]"
    Monitor        "Monitor[0]"
    DefaultDepth    24
    Option         "TVOutFormat" "Component" 
    Option         "TVStandard" "HD480i" 
    Option         "ConnectedMonitor" "TV"
    SubSection     "Display"
        Depth       24
        Modes      "1024x768_60" "720x576_60" "720x480_60" "640x480_60" "480x320_60"
    EndSubSection
EndSection

```

Note: it's morphed from an xorg.conf from when I set it up and had two screens, LCD and TV. I removed all the LCD stuff but left the server section, etc.
